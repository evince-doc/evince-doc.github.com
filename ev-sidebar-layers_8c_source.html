180 static void
181 ev_sidebar_layers_visibility_changed (GtkCellRendererToggle *cell,
182  gchar *path_str,
183  EvSidebarLayers *ev_layers)
184 {
185  GtkTreeModel *model;
186  GtkTreePath *path;
187  GtkTreeIter iter;
188  gboolean visible;
189  EvLayer *layer;
190 
191  model = gtk_tree_view_get_model (ev_layers->priv->tree_view);
192 
193  path = gtk_tree_path_new_from_string (path_str);
194  gtk_tree_model_get_iter (model, &iter, path);
195  gtk_tree_model_get (model, &iter,
196  EV_DOCUMENT_LAYERS_COLUMN_VISIBLE, &visible,
197  EV_DOCUMENT_LAYERS_COLUMN_LAYER, &layer,
198  -1);
199 
200  visible = !visible;
201  if (visible) {
202  gint rb_group;
203 
204  ev_document_layers_show_layer (EV_DOCUMENT_LAYERS (ev_layers->priv->document),
205  layer);
206 
207  rb_group = ev_layer_get_rb_group (layer);
208  if (rb_group) {
209  gtk_tree_model_foreach (model,
210  (GtkTreeModelForeachFunc)clear_rb_group,
211  &rb_group);
212  }
213  } else {
214  ev_document_layers_hide_layer (EV_DOCUMENT_LAYERS (ev_layers->priv->document),
215  layer);
216  }
217 
218  gtk_tree_store_set (GTK_TREE_STORE (model), &iter,
219  EV_DOCUMENT_LAYERS_COLUMN_VISIBLE, visible,
220  -1);
221 
222  if (ev_layer_is_parent (layer)) {
223  gtk_tree_model_foreach (model,
224  (GtkTreeModelForeachFunc)update_kids,
225  &iter);
226  }
227 
228  gtk_tree_path_free (path);
229 
230  g_signal_emit (ev_layers, signals[LAYERS_VISIBILITY_CHANGED], 0);
231 }

336 static void
337 update_layers_state (GtkTreeModel *model,
338  GtkTreeIter *iter,
339  EvDocumentLayers *document_layers)
340 {
341  EvLayer *layer;
342  gboolean visible;
343  GtkTreeIter child_iter;
344 
345  do {
346  gtk_tree_model_get (model, iter,
347  EV_DOCUMENT_LAYERS_COLUMN_VISIBLE, &visible,
348  EV_DOCUMENT_LAYERS_COLUMN_LAYER, &layer,
349  -1);
350  if (layer) {
351  gboolean layer_visible;
352 
353  layer_visible = ev_document_layers_layer_is_visible (document_layers, layer);
354  if (layer_visible != visible) {
355  gtk_tree_store_set (GTK_TREE_STORE (model), iter,
356  EV_DOCUMENT_LAYERS_COLUMN_VISIBLE, layer_visible,
357  -1);
358  }
359  }
360 
361  if (gtk_tree_model_iter_children (model, &child_iter, iter))
362  update_layers_state (model, &child_iter, document_layers);
363  } while (gtk_tree_model_iter_next (model, iter));
364 }

393 static void
394 ev_sidebar_layers_document_changed_cb (EvDocumentModel *model,
395  GParamSpec *pspec,
396  EvSidebarLayers *sidebar_layers)
397 {
398  EvDocument *document = ev_document_model_get_document (model);
399  EvSidebarLayersPrivate *priv = sidebar_layers->priv;
400 
401  if (!EV_IS_DOCUMENT_LAYERS (document))
402  return;
403 
404  if (priv->document) {
405  gtk_tree_view_set_model (GTK_TREE_VIEW (priv->tree_view), NULL);
406  g_object_unref (priv->document);
407  }
408 
409  priv->document = g_object_ref (document);
410 
411  if (priv->job) {
412  g_signal_handlers_disconnect_by_func (priv->job,
413  job_finished_callback,
414  sidebar_layers);
415  g_object_unref (priv->job);
416  }
417 
418  priv->job = ev_job_layers_new (document);
419  g_signal_connect (priv->job, "finished",
420  G_CALLBACK (job_finished_callback),
421  sidebar_layers);
422  /* The priority doesn't matter for this job */
423  ev_job_scheduler_push_job (priv->job, EV_JOB_PRIORITY_NONE);
424 }
