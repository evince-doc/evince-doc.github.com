104 static EvDocument *
105 evince_thumbnailer_get_document (GFile *file)
106 {
107  EvDocument *document = NULL;
108  gchar *uri;
109  GFile *tmp_file = NULL;
110  GError *error = NULL;
111 
112  if (!g_file_is_native (file)) {
113  gchar *base_name, *template;
114 
115  base_name = g_file_get_basename (file);
116  template = g_strdup_printf ("document.XXXXXX-%s", base_name);
117  g_free (base_name);
118 
119  tmp_file = ev_mkstemp_file (template, &error);
120  g_free (template);
121  if (!tmp_file) {
122  g_printerr ("Error loading remote document: %s\n", error->message);
123  g_error_free (error);
124 
125  return NULL;
126  }
127 
128  g_file_copy (file, tmp_file, G_FILE_COPY_OVERWRITE,
129  NULL, NULL, NULL, &error);
130  if (error) {
131  g_printerr ("Error loading remote document: %s\n", error->message);
132  g_error_free (error);
133  g_object_unref (tmp_file);
134 
135  return NULL;
136  }
137  uri = g_file_get_uri (tmp_file);
138  } else {
139  uri = g_file_get_uri (file);
140  }
141 
142  document = ev_document_factory_get_document (uri, &error);
143  if (tmp_file) {
144  if (document) {
145  g_object_weak_ref (G_OBJECT (document),
146  (GWeakNotify)delete_temp_file,
147  tmp_file);
148  } else {
149  ev_tmp_file_unlink (tmp_file);
150  g_object_unref (tmp_file);
151  }
152  }
153  g_free (uri);
154  if (error) {
155  if (error->domain == EV_DOCUMENT_ERROR &&
156  error->code == EV_DOCUMENT_ERROR_ENCRYPTED) {
157  /* FIXME: Create a thumb for cryp docs */
158  g_error_free (error);
159  return NULL;
160  }
161  g_printerr ("Error loading document: %s\n", error->message);
162  g_error_free (error);
163  return NULL;
164  }
165 
166  return document;
167 }

169 static gboolean
170 evince_thumbnail_pngenc_get (EvDocument *document, const char *thumbnail, int size)
171 {
172  EvRenderContext *rc;
173  double width, height;
174  GdkPixbuf *pixbuf;
175  EvPage *page;
176 
177  page = ev_document_get_page (document, 0);
178 
179  ev_document_get_page_size (document, 0, &width, &height);
180 
181  rc = ev_render_context_new (page, 0, size / MAX (height, width));
182  pixbuf = ev_document_get_thumbnail (document, rc);
183  g_object_unref (rc);
184  g_object_unref (page);
185 
186  if (pixbuf != NULL) {
187  if (gdk_pixbuf_save (pixbuf, thumbnail, "png", NULL, NULL)) {
188  g_object_unref (pixbuf);
189  return TRUE;
190  }
191 
192  g_object_unref (pixbuf);
193  }
194 
195  return FALSE;
196 }

222 int
223 main (int argc, char *argv[])
224 {
225  EvDocument *document;
226  GOptionContext *context;
227  const char *input;
228  const char *output;
229  GFile *file;
230  GError *error = NULL;
231 
232  setlocale (LC_ALL, "");
233 
234  context = g_option_context_new ("- GNOME Document Thumbnailer");
235  g_option_context_add_main_entries (context, goption_options, NULL);
236 
237  if (!g_option_context_parse (context, &argc, &argv, &error)) {
238  g_printerr ("%s\n", error->message);
239  g_error_free (error);
240  print_usage (context);
241  g_option_context_free (context);
242 
243  return -1;
244  }
245 
246  input = file_arguments ? file_arguments[0] : NULL;
247  output = input ? file_arguments[1] : NULL;
248  if (!input || !output) {
249  print_usage (context);
250  g_option_context_free (context);
251 
252  return -1;
253  }
254 
255  g_option_context_free (context);
256 
257  if (size < 1) {
258  g_printerr ("Size cannot be smaller than 1 pixel\n");
259  return -1;
260  }
261 
262  input = file_arguments[0];
263  output = file_arguments[1];
264 
265  if (!ev_init ())
266  return -1;
267 
268  file = g_file_new_for_commandline_arg (input);
269  document = evince_thumbnailer_get_document (file);
270  g_object_unref (file);
271 
272  if (!document) {
273  ev_shutdown ();
274  return -2;
275  }
276 
277  if (time_limit)
278  time_monitor_start (input);
279 
280  if (EV_IS_ASYNC_RENDERER (document)) {
281  struct AsyncData data;
282 
283  gtk_init (&argc, &argv);
284 
285  data.document = document;
286  data.output = output;
287  data.size = size;
288 
289  g_thread_new ("ThmbnlrAsyncRndr",
290  (GThreadFunc) evince_thumbnail_pngenc_get_async,
291  &data);
292 
293  gtk_main ();
294 
295  g_object_unref (document);
296  ev_shutdown ();
297 
298  return data.success ? 0 : -2;
299  }
300 
301  if (!evince_thumbnail_pngenc_get (document, output, size)) {
302  g_object_unref (document);
303  ev_shutdown ();
304  return -2;
305  }
306 
307  time_monitor_stop ();
308  g_object_unref (document);
309  ev_shutdown ();
310 
311  return 0;
312 }
