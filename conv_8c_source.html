28 size_t ar_conv_rune_to_utf8(wchar_t rune, char *out, size_t size)
29 {
30  if (size < 1)
31  return 0;
32  if (rune < 0x0080) {
33  *out++ = rune & 0x7F;
34  return 1;
35  }
36  if (rune < 0x0800 && size >= 2) {
37  *out++ = 0xC0 | ((rune >> 6) & 0x1F);
38  *out++ = 0x80 | (rune & 0x3F);
39  return 2;

43  rune = 0xFFFD;

47  return 3;
