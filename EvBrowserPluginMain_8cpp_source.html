136 NPError NPP_GetValue(NPP instance, NPPVariable variable, void *value)
137 {
138  switch (variable) {
139  case NPPVpluginNameString:
140  *((char **)value) = const_cast<char *>(EvBrowserPlugin::nameString());
141  return NPERR_NO_ERROR;
142  case NPPVpluginDescriptionString:
143  *((char **)value) = const_cast<char *>(EvBrowserPlugin::descriptionString());
144  return NPERR_NO_ERROR;
145  case NPPVpluginNeedsXEmbed:
146  *((NPBool *)value) = TRUE;
147  return NPERR_NO_ERROR;
148  case NPPVpluginScriptableNPObject: {
149  EvBrowserPlugin *plugin = pluginForInstance(instance);
150  if (!plugin)
151  return NPERR_INVALID_PLUGIN_ERROR;
152 
153  browser->retainobject(static_cast<NPObject *>(plugin));
154  *((NPObject **)value) = plugin;
155  return NPERR_NO_ERROR;
156  }
157  default:
158  return NPERR_INVALID_PARAM;
159  }
160 
161  return NPERR_GENERIC_ERROR;
162 }

188 NPError NP_Initialize(NPNetscapeFuncs *browserFuncs, NPPluginFuncs *pluginFuncs)
189 {
190  if (!browserFuncs || !pluginFuncs)
191  return NPERR_INVALID_FUNCTABLE_ERROR;
192 
193  if ((browserFuncs->version >> 8) > NP_VERSION_MAJOR)
194  return NPERR_INCOMPATIBLE_VERSION_ERROR;
195 
196  if (!ev_init())
197  return NPERR_GENERIC_ERROR;
198 
199  gtk_init(nullptr, nullptr);
200 
201  browser = browserFuncs;
202  initializePluginFuncs(pluginFuncs);
203 
204  GBytes *resourceData = g_resources_lookup_data("/org/gnome/evince/browser/ui/evince-browser.css", G_RESOURCE_LOOKUP_FLAGS_NONE, nullptr);
205  if (resourceData) {
206  GtkCssProvider *cssProvider = gtk_css_provider_new();
207 
208  gtk_css_provider_load_from_data(cssProvider, static_cast<const gchar *>(g_bytes_get_data(resourceData, nullptr)), g_bytes_get_size(resourceData), nullptr);
209  g_bytes_unref(resourceData);
210 
211  gtk_style_context_add_provider_for_screen(gdk_screen_get_default(), GTK_STYLE_PROVIDER(cssProvider), GTK_STYLE_PROVIDER_PRIORITY_APPLICATION);
212  g_object_unref(cssProvider);
213  }
214 
215  return NPERR_NO_ERROR;
216 }

261 const char *NP_GetMIMEDescription()
262 {
263  if (mimeDescription)
264  return mimeDescription.get();
265 
266  if (!ev_init())
267  return nullptr;
268 
269  GString *mimeDescriptionStr = g_string_new(nullptr);
270 
271  GList *typesInfo = ev_backends_manager_get_all_types_info();
272  for (GList *l = typesInfo; l; l = g_list_next(l)) {
273  EvTypeInfo *info = static_cast<EvTypeInfo *>(l->data);
274 
275  for (unsigned i = 0; info->mime_types[i]; ++i) {
276  const char *extensions = [](const char *mime) -> const char * {
277  for (unsigned i = 0; i < G_N_ELEMENTS(mimeExtensions); ++i) {
278  if (g_ascii_strcasecmp(mimeExtensions[i].mime, mime) == 0)
279  return mimeExtensions[i].extensions;
280  }
281 
282  return nullptr;
283  }(info->mime_types[i]);
284 
285  if (!extensions)
286  continue;
287 
288  g_string_append_printf(mimeDescriptionStr, "%s:%s:%s;",
289  info->mime_types[i],
290  extensions,
291  info->desc);
292  }
293  }
294  g_list_free(typesInfo);
295 
296  mimeDescription.reset(g_string_free(mimeDescriptionStr, FALSE));
297 
298  ev_shutdown();
299 
300  return mimeDescription.get();
301 }
