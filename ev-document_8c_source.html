197 static void
198 ev_document_setup_cache (EvDocument *document)
199 {
200  EvDocumentPrivate *priv = document->priv;
201  gint i;
202 
203  /* Cache some info about the document to avoid
204  * going to the backends since it requires locks
205  */
206  priv->info = _ev_document_get_info (document);
207  priv->n_pages = _ev_document_get_n_pages (document);
208 
209  for (i = 0; i < priv->n_pages; i++) {
210  EvPage *page = ev_document_get_page (document, i);
211  gdouble page_width = 0;
212  gdouble page_height = 0;
213  EvPageSize *page_size;
214  gchar *page_label;
215 
216  _ev_document_get_page_size (document, page, &page_width, &page_height);
217 
218  if (i == 0) {
219  priv->uniform_width = page_width;
220  priv->uniform_height = page_height;
221  priv->max_width = priv->uniform_width;
222  priv->max_height = priv->uniform_height;
223  priv->min_width = priv->uniform_width;
224  priv->min_height = priv->uniform_height;
225  } else if (priv->uniform &&
226  (priv->uniform_width != page_width ||
227  priv->uniform_height != page_height)) {
228  /* It's a different page size. Backfill the array. */
229  int j;
230 
231  priv->page_sizes = g_new0 (EvPageSize, priv->n_pages);
232 
233  for (j = 0; j < i; j++) {
234  page_size = &(priv->page_sizes[j]);
235  page_size->width = priv->uniform_width;
236  page_size->height = priv->uniform_height;
237  }
238  priv->uniform = FALSE;
239  }
240  if (!priv->uniform) {
241  page_size = &(priv->page_sizes[i]);
242 
243  page_size->width = page_width;
244  page_size->height = page_height;
245 
246  if (page_width > priv->max_width)
247  priv->max_width = page_width;
248  if (page_width < priv->min_width)
249  priv->min_width = page_width;
250 
251  if (page_height > priv->max_height)
252  priv->max_height = page_height;
253  if (page_height < priv->min_height)
254  priv->min_height = page_height;
255  }
256 
257  page_label = _ev_document_get_page_label (document, page);
258  if (page_label) {
259  if (!priv->page_labels)
260  priv->page_labels = g_new0 (gchar *, priv->n_pages);
261 
262  priv->page_labels[i] = page_label;
263  priv->max_label = MAX (priv->max_label,
264  g_utf8_strlen (page_label, 256));
265  }
266 
267  g_object_unref (page);
268  }
269 }

496 ev_document_synctex_backward_search (EvDocument *document,
497  gint page_index,
498  gfloat x,
499  gfloat y)
500 {
501  EvSourceLink *result = NULL;

503 
504  g_return_val_if_fail (EV_IS_DOCUMENT (document), NULL);
505 
506  scanner = document->priv->synctex_scanner;
507  if (!scanner)
508  return NULL;
509 
510  if (synctex_edit_query (scanner, page_index + 1, x, y) > 0) {
511  synctex_node_t node;

519

521  result = ev_source_link_new (filename,
522  synctex_node_line (node),
523  synctex_node_column (node));
524  }
525  }
526  }

822 ev_document_find_page_by_label (EvDocument *document,
823  const gchar *page_label,
824  gint *page_index)
825 {
826  gint i, page;
827  glong value;
828  gchar *endptr = NULL;
829  EvDocumentPrivate *priv = document->priv;
830 
831  g_return_val_if_fail (EV_IS_DOCUMENT (document), FALSE);
832  g_return_val_if_fail (page_label != NULL, FALSE);
833  g_return_val_if_fail (page_index != NULL, FALSE);
834 
835  /* First, look for a literal label match */
836  for (i = 0; priv->page_labels && i < priv->n_pages; i ++) {
837  if (priv->page_labels[i] != NULL &&
838  ! strcmp (page_label, priv->page_labels[i])) {
839  *page_index = i;
840  return TRUE;
841  }
842  }
843 
844  /* Second, look for a match with case insensitively */
845  for (i = 0; priv->page_labels && i < priv->n_pages; i++) {
846  if (priv->page_labels[i] != NULL &&
847  ! strcasecmp (page_label, priv->page_labels[i])) {
848  *page_index = i;
849  return TRUE;
850  }
851  }
852 
853  /* Next, parse the label, and see if the number fits */
854  value = strtol (page_label, &endptr, 10);
855  if (endptr[0] == '\0') {
856  /* Page number is an integer */
857  page = MIN (G_MAXINT, value);
858 
859  /* convert from a page label to a page offset */
860  page --;
861  if (page >= 0 && page < priv->n_pages) {
862  *page_index = page;
863  return TRUE;
864  }
865  }
866 
867  return FALSE;
868 }
