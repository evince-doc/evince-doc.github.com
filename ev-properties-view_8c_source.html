111 /* This is cut out of gconvert.c from glib (and mildly modified). Not all
112  backends give valid UTF-8 for properties, so we make sure that is.
113  */
114 static gchar *
115 make_valid_utf8 (const gchar *name)
116 {
117  GString *string;
118  const gchar *remainder, *invalid;
119  gint remaining_bytes, valid_bytes;
120 
121  string = NULL;
122  remainder = name;
123  remaining_bytes = strlen (name);
124 
125  while (remaining_bytes != 0)
126  {
127  if (g_utf8_validate (remainder, remaining_bytes, &invalid))
128  break;
129  valid_bytes = invalid - remainder;
130 
131  if (string == NULL)
132  string = g_string_sized_new (remaining_bytes);
133 
134  g_string_append_len (string, remainder, valid_bytes);
135  g_string_append_c (string, '?');
136 
137  remaining_bytes -= valid_bytes + 1;
138  remainder = invalid + 1;
139  }
140 
141  if (string == NULL)
142  return g_strdup (name);
143 
144  g_string_append (string, remainder);
145 
146  g_assert (g_utf8_validate (string->str, -1, NULL));
147 
148  return g_string_free (string, FALSE);
149 }

151 static void
152 set_property (EvPropertiesView *properties,
153  GtkGrid *grid,
154  Property property,
155  const gchar *text,
156  gint *row)
157 {
158  GtkWidget *label;
159  gchar *markup;
160  gchar *valid_text;
161 
162  if (!properties->labels[property]) {
163  label = gtk_label_new (NULL);
164  g_object_set (G_OBJECT (label), "xalign", 0.0, NULL);
165  markup = g_strdup_printf ("<b>%s</b>", _(properties_info[property].label));
166  gtk_label_set_markup (GTK_LABEL (label), markup);
167  g_free (markup);
168 
169  gtk_grid_attach (grid, label, 0, *row, 1, 1);
170  gtk_widget_show (label);
171  }
172 
173  if (!properties->labels[property]) {
174  label = gtk_label_new (NULL);
175 
176  g_object_set (G_OBJECT (label),
177  "xalign", 0.0,
178  "width_chars", 25,
179  "selectable", TRUE,
180  "ellipsize", PANGO_ELLIPSIZE_END,
181  NULL);
182  } else {
183  label = properties->labels[property];
184  }
185 
186  if (text == NULL || text[0] == '\000') {
187  /* translators: This is used when a document property does
188  not have a value. Examples:
189  Author: None
190  Keywords: None
191  */
192  markup = g_markup_printf_escaped ("<i>%s</i>", _("None"));
193  gtk_label_set_markup (GTK_LABEL (label), markup);
194  g_free (markup);
195  } else {
196  valid_text = make_valid_utf8 (text ? text : "");
197  gtk_label_set_text (GTK_LABEL (label), valid_text);
198  g_free (valid_text);
199  }
200 
201  if (!properties->labels[property]) {
202  gtk_grid_attach (grid, label, 1, *row, 1, 1);
203  properties->labels[property] = label;
204  }
205 
206  gtk_widget_show (label);
207 
208  *row += 1;
209 }

211 static GtkUnit
212 get_default_user_units (void)
213 {
214  /* Translate to the default units to use for presenting
215  * lengths to the user. Translate to default:inch if you
216  * want inches, otherwise translate to default:mm.
217  * Do *not* translate it to "predefinito:mm", if it
218  * it isn't default:mm or default:inch it will not work
219  */
220  gchar *e = _("default:mm");
221 
222 #ifdef HAVE__NL_MEASUREMENT_MEASUREMENT
223  gchar *imperial = NULL;
224 
225  imperial = nl_langinfo (_NL_MEASUREMENT_MEASUREMENT);
226  if (imperial && imperial[0] == 2)
227  return GTK_UNIT_INCH; /* imperial */
228  if (imperial && imperial[0] == 1)
229  return GTK_UNIT_MM; /* metric */
230 #endif
231 
232  if (strcmp (e, "default:mm") == 0)
233  return GTK_UNIT_MM;
234  if (strcmp (e, "default:inch") == 0)
235  return GTK_UNIT_INCH;
236 
237  g_warning ("Whoever translated default:mm did so wrongly.\n");
238 
239  return GTK_UNIT_MM;
240 }

242 static gdouble
243 get_tolerance (gdouble size)
244 {
245  if (size < 150.0f)
246  return 1.5f;
247  else if (size >= 150.0f && size <= 600.0f)
248  return 2.0f;
249  else
250  return 3.0f;
251 }

253 static char *
254 ev_regular_paper_size (const EvDocumentInfo *info)
255 {
256  GList *paper_sizes, *l;
257  gchar *exact_size;
258  gchar *str = NULL;
259  GtkUnit units;
260 
261  units = get_default_user_units ();
262 
263  if (units == GTK_UNIT_MM) {
264  exact_size = g_strdup_printf(_("%.0f × %.0f mm"),
265  info->paper_width,
266  info->paper_height);
267  } else {
268  exact_size = g_strdup_printf (_("%.2f × %.2f inch"),
269  info->paper_width / 25.4f,
270  info->paper_height / 25.4f);
271  }
272 
273  paper_sizes = gtk_paper_size_get_paper_sizes (FALSE);
274 
275  for (l = paper_sizes; l && l->data; l = g_list_next (l)) {
276  GtkPaperSize *size = (GtkPaperSize *) l->data;
277  gdouble paper_width;
278  gdouble paper_height;
279  gdouble width_tolerance;
280  gdouble height_tolerance;
281 
282  paper_width = gtk_paper_size_get_width (size, GTK_UNIT_MM);
283  paper_height = gtk_paper_size_get_height (size, GTK_UNIT_MM);
284 
285  width_tolerance = get_tolerance (paper_width);
286  height_tolerance = get_tolerance (paper_height);
287 
288  if (ABS (info->paper_height - paper_height) <= height_tolerance &&
289  ABS (info->paper_width - paper_width) <= width_tolerance) {
290  /* Note to translators: first placeholder is the paper name (eg.
291  * A4), second placeholder is the paper size (eg. 297x210 mm) */
292  str = g_strdup_printf (_("%s, Portrait (%s)"),
293  gtk_paper_size_get_display_name (size),
294  exact_size);
295  } else if (ABS (info->paper_width - paper_height) <= height_tolerance &&
296  ABS (info->paper_height - paper_width) <= width_tolerance) {
297  /* Note to translators: first placeholder is the paper name (eg.
298  * A4), second placeholder is the paper size (eg. 297x210 mm) */
299  str = g_strdup_printf ( _("%s, Landscape (%s)"),
300  gtk_paper_size_get_display_name (size),
301  exact_size);
302  }
303  }
304 
305  g_list_foreach (paper_sizes, (GFunc) gtk_paper_size_free, NULL);
306  g_list_free (paper_sizes);
307 
308  if (str != NULL) {
309  g_free (exact_size);
310  return str;
311  }
312 
313  return exact_size;
314 }

316 void
317 ev_properties_view_set_info (EvPropertiesView *properties, const EvDocumentInfo *info)
318 {
319  GtkWidget *grid;
320  gchar *text;
321  gint row = 0;
322 
323  grid = properties->grid;
324 
325  if (info->fields_mask & EV_DOCUMENT_INFO_TITLE) {
326  set_property (properties, GTK_GRID (grid), TITLE_PROPERTY, info->title, &row);
327  }
328  set_property (properties, GTK_GRID (grid), URI_PROPERTY, properties->uri, &row);
329  if (info->fields_mask & EV_DOCUMENT_INFO_SUBJECT) {
330  set_property (properties, GTK_GRID (grid), SUBJECT_PROPERTY, info->subject, &row);
331  }
332  if (info->fields_mask & EV_DOCUMENT_INFO_AUTHOR) {
333  set_property (properties, GTK_GRID (grid), AUTHOR_PROPERTY, info->author, &row);
334  }
335  if (info->fields_mask & EV_DOCUMENT_INFO_KEYWORDS) {
336  set_property (properties, GTK_GRID (grid), KEYWORDS_PROPERTY, info->keywords, &row);
337  }
338  if (info->fields_mask & EV_DOCUMENT_INFO_PRODUCER) {
339  set_property (properties, GTK_GRID (grid), PRODUCER_PROPERTY, info->producer, &row);
340  }
341  if (info->fields_mask & EV_DOCUMENT_INFO_CREATOR) {
342  set_property (properties, GTK_GRID (grid), CREATOR_PROPERTY, info->creator, &row);
343  }
344  if (info->fields_mask & EV_DOCUMENT_INFO_CREATION_DATE) {
345  text = ev_document_misc_format_date (info->creation_date);
346  set_property (properties, GTK_GRID (grid), CREATION_DATE_PROPERTY, text, &row);
347  g_free (text);
348  }
349  if (info->fields_mask & EV_DOCUMENT_INFO_MOD_DATE) {
350  text = ev_document_misc_format_date (info->modified_date);
351  set_property (properties, GTK_GRID (grid), MOD_DATE_PROPERTY, text, &row);
352  g_free (text);
353  }
354  if (info->fields_mask & EV_DOCUMENT_INFO_FORMAT) {
355  set_property (properties, GTK_GRID (grid), FORMAT_PROPERTY, info->format, &row);
356  }
357  if (info->fields_mask & EV_DOCUMENT_INFO_N_PAGES) {
358  text = g_strdup_printf ("%d", info->n_pages);
359  set_property (properties, GTK_GRID (grid), N_PAGES_PROPERTY, text, &row);
360  g_free (text);
361  }
362  if (info->fields_mask & EV_DOCUMENT_INFO_LINEARIZED) {
363  set_property (properties, GTK_GRID (grid), LINEARIZED_PROPERTY, info->linearized, &row);
364  }
365  if (info->fields_mask & EV_DOCUMENT_INFO_SECURITY) {
366  set_property (properties, GTK_GRID (grid), SECURITY_PROPERTY, info->security, &row);
367  }
368  if (info->fields_mask & EV_DOCUMENT_INFO_PAPER_SIZE) {
369  text = ev_regular_paper_size (info);
370  set_property (properties, GTK_GRID (grid), PAPER_SIZE_PROPERTY, text, &row);
371  g_free (text);
372  }
373 }
