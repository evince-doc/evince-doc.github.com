148 static gchar *
149 comics_regex_quote (const gchar *unquoted_string)
150 {
151  const gchar *p;
152  GString *dest;
153 
154  dest = g_string_new ("'");
155 
156  p = unquoted_string;
157 
158  while (*p) {
159  switch (*p) {
160  /* * matches a sequence of 0 or more characters */
161  case ('*'):
162  /* ? matches exactly 1 charactere */
163  case ('?'):
164  /* [...] matches any single character found inside
165  * the brackets. Disabling the first bracket is enough.
166  */
167  case ('['):
168  g_string_append (dest, "[");
169  g_string_append_c (dest, *p);
170  g_string_append (dest, "]");
171  break;
172  /* Because \ escapes regex expressions that we are
173  * disabling for unzip, we need to disable \ too */
174  case ('\\'):
175  g_string_append (dest, "[\\\\]");
176  break;
177  /* Escape single quote inside the string */
178  case ('\''):
179  g_string_append (dest, "'\\''");
180  break;
181  default:
182  g_string_append_c (dest, *p);
183  break;
184  }
185  ++p;
186  }
187  g_string_append_c (dest, '\'');
188  return g_string_free (dest, FALSE);
189 }

229  } else {
230  g_set_error (error,
231  EV_DOCUMENT_ERROR,
232  EV_DOCUMENT_ERROR_INVALID,
233  _("The command “%s” did not end normally."),
234  basename);
235  g_free (std_out);

237  g_free (basename);

289  return FALSE;

291  return TRUE;
292  } else {
293  g_free (quoted_file);
294  g_free (quoted_command);

300 /* This function chooses an external command for decompressing a comic
301  * book based on its mime tipe. */
302 static gboolean
303 comics_check_decompress_command (gchar *mime_type,
304  ComicsDocument *comics_document,
305  GError **error)
306 {
307  gboolean success;
308  gchar *std_out, *std_err;
309  gint retval;
310  GError *err = NULL;
311 
312  /* FIXME, use proper cbr/cbz mime types once they're
313  * included in shared-mime-info */
314 
315  if (!strcmp (mime_type, "application/x-cbr") ||
316  !strcmp (mime_type, "application/x-rar")) {
317  /* The RARLAB provides a no-charge proprietary (freeware)
318  * decompress-only client for Linux called unrar. Another
319  * option is a GPLv2-licensed command-line tool developed by
320  * the Gna! project. Confusingly enough, the free software RAR
321  * decoder is also named unrar. For this reason we need to add
322  * some lines for disambiguation. Sorry for the added the
323  * complexity but it's life :)
324  * Finally, some distributions, like Debian, rename this free
325  * option as unrar-free.
326  * */
327  comics_document->selected_command =
328  g_find_program_in_path ("unrar");
329  if (comics_document->selected_command) {
330  /* We only use std_err to avoid printing useless error
331  * messages on the terminal */
332  success =
333  g_spawn_command_line_sync (
334  comics_document->selected_command,
335  &std_out, &std_err,
336  &retval, &err);
337  if (!success) {
338  g_propagate_error (error, err);
339  g_error_free (err);
340  return FALSE;
341  /* I don't check retval status because RARLAB unrar
342  * doesn't have a way to return 0 without involving an
343  * operation with a file*/
344  } else if (WIFEXITED (retval)) {
345  if (g_strrstr (std_out,"freeware") != NULL)
346  /* The RARLAB freeware client */
347  comics_document->command_usage = RARLABS;
348  else
349  /* The Gna! free software client */
350  comics_document->command_usage = GNAUNRAR;
351 
352  g_free (std_out);
353  g_free (std_err);
354  return TRUE;
355  }
356  }
357  /* The Gna! free software client with Debian naming convention */
358  comics_document->selected_command =
359  g_find_program_in_path ("unrar-free");
360  if (comics_document->selected_command) {
361  comics_document->command_usage = GNAUNRAR;
362  return TRUE;
363  }
364 
365  } else if (!strcmp (mime_type, "application/x-cbz") ||
366  !strcmp (mime_type, "application/zip")) {
367  /* InfoZIP's unzip program */
368  comics_document->selected_command =
369  g_find_program_in_path ("unzip");
370  comics_document->alternative_command =
371  g_find_program_in_path ("zipnote");
372  if (comics_document->selected_command &&
373  comics_document->alternative_command) {
374  comics_document->command_usage = UNZIP;
375  return TRUE;
376  }
377  /* fallback mode using 7za and 7z from p7zip project */
378  comics_document->selected_command =
379  g_find_program_in_path ("7za");
380  if (comics_document->selected_command) {
381  comics_document->command_usage = P7ZIP;
382  return TRUE;
383  }
384  comics_document->selected_command =
385  g_find_program_in_path ("7z");
386  if (comics_document->selected_command) {
387  comics_document->command_usage = P7ZIP;
388  return TRUE;
389  }
390 
391  } else if (!strcmp (mime_type, "application/x-cb7") ||
392  !strcmp (mime_type, "application/x-7z-compressed")) {
393  /* 7zr, 7za and 7z are the commands from the p7zip project able
394  * to decompress .7z files */
395  comics_document->selected_command =
396  g_find_program_in_path ("7zr");
397  if (comics_document->selected_command) {
398  comics_document->command_usage = P7ZIP;
399  return TRUE;
400  }
401  comics_document->selected_command =
402  g_find_program_in_path ("7za");
403  if (comics_document->selected_command) {
404  comics_document->command_usage = P7ZIP;
405  return TRUE;
406  }
407  comics_document->selected_command =
408  g_find_program_in_path ("7z");
409  if (comics_document->selected_command) {
410  comics_document->command_usage = P7ZIP;
411  return TRUE;
412  }
413  } else if (!strcmp (mime_type, "application/x-cbt") ||
414  !strcmp (mime_type, "application/x-tar")) {
415  /* tar utility (Tape ARchive) */
416  comics_document->selected_command =
417  g_find_program_in_path ("tar");
418  if (comics_document->selected_command) {
419  comics_document->command_usage = TAR;
420  return TRUE;
421  }
422  } else {
423  g_set_error (error,
424  EV_DOCUMENT_ERROR,
425  EV_DOCUMENT_ERROR_INVALID,
426  _("Not a comic book MIME type: %s"),
427  mime_type);
428  return FALSE;
429  }
430  g_set_error_literal (error,
431  EV_DOCUMENT_ERROR,
432  EV_DOCUMENT_ERROR_INVALID,
433  _("Can't find an appropriate command to "
434  "decompress this type of comic book"));
435  return FALSE;
436 }

445 static gboolean
446 comics_document_load (EvDocument *document,
447  const char *uri,
448  GError **error)
449 {
450  ComicsDocument *comics_document = COMICS_DOCUMENT (document);
451  GSList *supported_extensions;
452  gchar *std_out;
453  gchar *mime_type;
454  gchar **cb_files, *cb_file;
455  gboolean success;
456  int i, retval;
457  GError *err = NULL;
458 
459  comics_document->archive = g_filename_from_uri (uri, NULL, error);
460  if (!comics_document->archive)
461  return FALSE;
462 
463  mime_type = ev_file_get_mime_type (uri, FALSE, &err);
464  if (mime_type == NULL)
465  return FALSE;
466 
467  if (!comics_check_decompress_command (mime_type, comics_document,
468  error)) {
469  g_free (mime_type);
470  return FALSE;
471  } else if (!comics_generate_command_lines (comics_document, error)) {
472  g_free (mime_type);
473  return FALSE;
474  }
475 
476  g_free (mime_type);
477 
478  /* Get list of files in archive */
479  success = g_spawn_command_line_sync (comics_document->list_command,
480  &std_out, NULL, &retval, error);
481 
482  if (!success) {
483  return FALSE;
484  } else if (!WIFEXITED(retval) || WEXITSTATUS(retval) != EXIT_SUCCESS) {
485  g_set_error_literal (error,
486  EV_DOCUMENT_ERROR,
487  EV_DOCUMENT_ERROR_INVALID,
488  _("File corrupted"));
489  return FALSE;
490  }
491 
492  /* FIXME: is this safe against filenames containing \n in the archive ? */
493  cb_files = g_strsplit (std_out, EV_EOL, 0);
494 
495  g_free (std_out);
496 
497  if (!cb_files) {
498  g_set_error_literal (error,
499  EV_DOCUMENT_ERROR,
500  EV_DOCUMENT_ERROR_INVALID,
501  _("No files in archive"));
502  return FALSE;
503  }
504 
505  comics_document->page_names = g_ptr_array_sized_new (64);
506 
507  supported_extensions = get_supported_image_extensions ();
508  for (i = 0; cb_files[i] != NULL; i++) {
509  if (comics_document->offset != NO_OFFSET) {
510  if (g_utf8_strlen (cb_files[i],-1) >
511  comics_document->offset) {
512  cb_file =
513  g_utf8_offset_to_pointer (cb_files[i],
514  comics_document->offset);
515  } else {
516  continue;
517  }
518  } else {
519  cb_file = cb_files[i];
520  }
521  gchar *suffix = g_strrstr (cb_file, ".");
522  if (!suffix)
523  continue;
524  suffix = g_ascii_strdown (suffix + 1, -1);
525  if (g_slist_find_custom (supported_extensions, suffix,
526  (GCompareFunc) strcmp) != NULL) {
527  g_ptr_array_add (comics_document->page_names,
528  g_strstrip (g_strdup (cb_file)));
529  }
530  g_free (suffix);
531  }
532  g_strfreev (cb_files);
533  g_slist_foreach (supported_extensions, (GFunc) g_free, NULL);
534  g_slist_free (supported_extensions);
535 
536  if (comics_document->page_names->len == 0) {
537  g_set_error (error,
538  EV_DOCUMENT_ERROR,
539  EV_DOCUMENT_ERROR_INVALID,
540  _("No images found in archive %s"),
541  uri);
542  return FALSE;
543  }
544 
545  /* Now sort the pages */
546  g_ptr_array_sort (comics_document->page_names, sort_page_names);
547 
548  return TRUE;
549 }

573 static void
574 comics_document_get_page_size (EvDocument *document,
575  EvPage *page,
576  double *width,
577  double *height)
578 {
579  GdkPixbufLoader *loader;
580  char **argv;
581  guchar buf[1024];
582  gboolean success, got_size = FALSE;
583  gint outpipe = -1;
584  GPid child_pid;
585  gssize bytes;
586  GdkPixbuf *pixbuf;
587  gchar *filename;
588  ComicsDocument *comics_document = COMICS_DOCUMENT (document);
589 
590  if (!comics_document->decompress_tmp) {
591  argv = extract_argv (document, page->index);
592  success = g_spawn_async_with_pipes (NULL, argv, NULL,
593  G_SPAWN_SEARCH_PATH |
594  G_SPAWN_STDERR_TO_DEV_NULL,
595  NULL, NULL,
596  &child_pid,
597  NULL, &outpipe, NULL, NULL);
598  g_strfreev (argv);
599  g_return_if_fail (success == TRUE);
600 
601  loader = gdk_pixbuf_loader_new ();
602  g_signal_connect (loader, "area-prepared",
603  G_CALLBACK (get_page_size_area_prepared_cb),
604  &got_size);
605 
606  while (outpipe >= 0) {
607  bytes = read (outpipe, buf, 1024);
608 
609  if (bytes > 0)
610  gdk_pixbuf_loader_write (loader, buf, bytes, NULL);
611  if (bytes <= 0 || got_size) {
612  close (outpipe);
613  outpipe = -1;
614  gdk_pixbuf_loader_close (loader, NULL);
615  }
616  }
617  pixbuf = gdk_pixbuf_loader_get_pixbuf (loader);
618  if (pixbuf) {
619  if (width)
620  *width = gdk_pixbuf_get_width (pixbuf);
621  if (height)
622  *height = gdk_pixbuf_get_height (pixbuf);
623  }
624  g_spawn_close_pid (child_pid);
625  g_object_unref (loader);
626  } else {
627  filename = g_build_filename (comics_document->dir,
628  (char *) comics_document->page_names->pdata[page->index],
629  NULL);
630  pixbuf = gdk_pixbuf_new_from_file (filename, NULL);
631  if (pixbuf) {
632  if (width)
633  *width = gdk_pixbuf_get_width (pixbuf);
634  if (height)
635  *height = gdk_pixbuf_get_height (pixbuf);
636  g_object_unref (pixbuf);
637  }
638  g_free (filename);
639  }
640 }
