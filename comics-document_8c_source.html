88  }
89

91  g_ptr_array_free (array, TRUE);
92  } else {
93  g_ptr_array_add (array, NULL);
94  ret = (char **) g_ptr_array_free (array, FALSE);
95  }
96 
97 out:
98  ev_archive_reset (comics_document->archive);
99  return ret;
100 }

102 /* This function chooses the archive decompression support
103  * book based on its mime type. */
104 static gboolean
105 comics_check_decompress_support (gchar *mime_type,
106  ComicsDocument *comics_document,
107  GError **error)
108 {
109  if (g_content_type_is_a (mime_type, "application/x-cbr") ||
110  g_content_type_is_a (mime_type, "application/x-rar")) {
111  if (ev_archive_set_archive_type (comics_document->archive, EV_ARCHIVE_TYPE_RAR))
112  return TRUE;
113  } else if (g_content_type_is_a (mime_type, "application/x-cbz") ||
114  g_content_type_is_a (mime_type, "application/zip")) {
115  if (ev_archive_set_archive_type (comics_document->archive, EV_ARCHIVE_TYPE_ZIP))
116  return TRUE;
117  } else if (g_content_type_is_a (mime_type, "application/x-cb7") ||
118  g_content_type_is_a (mime_type, "application/x-7z-compressed")) {
119  if (ev_archive_set_archive_type (comics_document->archive, EV_ARCHIVE_TYPE_7Z))
120  return TRUE;
121  } else if (g_content_type_is_a (mime_type, "application/x-cbt") ||
122  g_content_type_is_a (mime_type, "application/x-tar")) {
123  if (ev_archive_set_archive_type (comics_document->archive, EV_ARCHIVE_TYPE_TAR))
124  return TRUE;
125  } else {
126  g_set_error (error,
127  EV_DOCUMENT_ERROR,
128  EV_DOCUMENT_ERROR_INVALID,
129  _("Not a comic book MIME type: %s"),
130  mime_type);
131  return FALSE;
132  }
133  g_set_error_literal (error,
134  EV_DOCUMENT_ERROR,
135  EV_DOCUMENT_ERROR_INVALID,
136  _("libarchive lacks support for this comic book’s "
137  "compression, please contact your distributor"));
138  return FALSE;
139 }

159 static gboolean
160 comics_document_load (EvDocument *document,
161  const char *uri,
162  GError **error)
163 {
164  ComicsDocument *comics_document = COMICS_DOCUMENT (document);
165  GSList *supported_extensions;
166  gchar *mime_type;
167  gchar **cb_files, *cb_file;
168  int i;
169  GError *err = NULL;
170  GFile *file;
171 
172  file = g_file_new_for_uri (uri);
173  comics_document->archive_path = g_file_get_path (file);
174  g_object_unref (file);
175 
176  if (!comics_document->archive_path) {
177  g_set_error_literal (error,
178  EV_DOCUMENT_ERROR,
179  EV_DOCUMENT_ERROR_INVALID,
180  _("Can not get local path for archive"));
181  return FALSE;
182  }
183 
184  comics_document->archive_uri = g_strdup (uri);
185 
186  mime_type = ev_file_get_mime_type (uri, FALSE, &err);
187  if (mime_type == NULL)
188  return FALSE;
189 
190  if (!comics_check_decompress_support (mime_type, comics_document, error)) {
191  g_free (mime_type);
192  return FALSE;
193  }
194  g_free (mime_type);
195 
196  /* Get list of files in archive */
197  cb_files = comics_document_list (comics_document);
198  if (!cb_files) {
199  g_set_error_literal (error,
200  EV_DOCUMENT_ERROR,
201  EV_DOCUMENT_ERROR_INVALID,
202  _("File corrupted or no files in archive"));
203  return FALSE;
204  }
205 
206  comics_document->page_names = g_ptr_array_sized_new (64);
207 
208  supported_extensions = get_supported_image_extensions ();
209  for (i = 0; cb_files[i] != NULL; i++) {
210  cb_file = cb_files[i];
211  gchar *suffix = g_strrstr (cb_file, ".");
212  if (!suffix)
213  continue;
214  suffix = g_ascii_strdown (suffix + 1, -1);
215  if (g_slist_find_custom (supported_extensions, suffix,
216  (GCompareFunc) strcmp) != NULL) {
217  g_ptr_array_add (comics_document->page_names,
218  g_strstrip (g_strdup (cb_file)));
219  }
220  g_free (suffix);
221  }
222  g_strfreev (cb_files);
223  g_slist_foreach (supported_extensions, (GFunc) g_free, NULL);
224  g_slist_free (supported_extensions);
225 
226  if (comics_document->page_names->len == 0) {
227  g_set_error (error,
228  EV_DOCUMENT_ERROR,
229  EV_DOCUMENT_ERROR_INVALID,
230  _("No images found in archive %s"),
231  uri);
232  return FALSE;
233  }
234 
235  /* Now sort the pages */
236  g_ptr_array_sort (comics_document->page_names, sort_page_names);
237 
238  return TRUE;
239 }

279 static void
280 comics_document_get_page_size (EvDocument *document,
281  EvPage *page,
282  double *width,
283  double *height)
284 {
285  GdkPixbufLoader *loader;
286  ComicsDocument *comics_document = COMICS_DOCUMENT (document);
287  const char *page_path;
288  PixbufInfo info;
289  GError *error = NULL;
290 
291  if (!ev_archive_open_filename (comics_document->archive, comics_document->archive_path, &error)) {
292  g_warning ("Fatal error opening archive: %s", error->message);
293  g_error_free (error);
294  goto out;
295  }
296 
297  loader = gdk_pixbuf_loader_new ();
298  info.got_info = FALSE;
299  g_signal_connect (loader, "size-prepared",
300  G_CALLBACK (get_page_size_prepared_cb),
301  &info);
302 
303  page_path = g_ptr_array_index (comics_document->page_names, page->index);
304 
305  while (1) {
306  const char *name;
307  GError *error = NULL;
308 
309  if (!ev_archive_read_next_header (comics_document->archive, &error)) {
310  if (error != NULL) {
311  g_warning ("Fatal error handling archive: %s", error->message);
312  g_error_free (error);
313  }
314  break;
315  }
316 
317  name = ev_archive_get_entry_pathname (comics_document->archive);
318  if (g_strcmp0 (name, page_path) == 0) {
319  char buf[BLOCK_SIZE];
320  gssize read;
321 
322  read = ev_archive_read_data (comics_document->archive, buf, sizeof(buf), &error);
323  while (read > 0 && !info.got_info) {
324  gdk_pixbuf_loader_write (loader, (guchar *) buf, read, NULL);
325  read = ev_archive_read_data (comics_document->archive, buf, BLOCK_SIZE, &error);
326  }
327  if (read < 0) {
328  g_warning ("Fatal error reading '%s' in archive: %s", name, error->message);
329  g_error_free (error);
330  }
331  break;
332  }
333  }
334 
335  gdk_pixbuf_loader_close (loader, NULL);
336  g_object_unref (loader);
337 
338  if (info.got_info) {
339  if (width)
340  *width = info.width;
341  if (height)
342  *height = info.height;
343  }
344 
345 out:
346  ev_archive_reset (comics_document->archive);
347 }

361 static GdkPixbuf *
362 comics_document_render_pixbuf (EvDocument *document,
363  EvRenderContext *rc)
364 {
365  GdkPixbufLoader *loader;
366  GdkPixbuf *tmp_pixbuf;
367  GdkPixbuf *rotated_pixbuf = NULL;
368  ComicsDocument *comics_document = COMICS_DOCUMENT (document);
369  const char *page_path;
370  GError *error = NULL;
371 
372  if (!ev_archive_open_filename (comics_document->archive, comics_document->archive_path, &error)) {
373  g_warning ("Fatal error opening archive: %s", error->message);
374  g_error_free (error);
375  goto out;
376  }
377 
378  loader = gdk_pixbuf_loader_new ();
379  g_signal_connect (loader, "size-prepared",
380  G_CALLBACK (render_pixbuf_size_prepared_cb),
381  rc);
382 
383  page_path = g_ptr_array_index (comics_document->page_names, rc->page->index);
384 
385  while (1) {
386  const char *name;
387 
388  if (!ev_archive_read_next_header (comics_document->archive, &error)) {
389  if (error != NULL) {
390  g_warning ("Fatal error handling archive: %s", error->message);
391  g_error_free (error);
392  }
393  break;
394  }
395 
396  name = ev_archive_get_entry_pathname (comics_document->archive);
397  if (g_strcmp0 (name, page_path) == 0) {
398  size_t size = ev_archive_get_entry_size (comics_document->archive);
399  char *buf;
400  ssize_t read;
401 
402  buf = g_malloc (size);
403  read = ev_archive_read_data (comics_document->archive, buf, size, &error);
404  if (read <= 0) {
405  if (read < 0) {
406  g_warning ("Fatal error reading '%s' in archive: %s", name, error->message);
407  g_error_free (error);
408  } else {
409  g_warning ("Read an empty file from the archive");
410  }
411  } else {
412  gdk_pixbuf_loader_write (loader, (guchar *) buf, size, NULL);
413  }
414  g_free (buf);
415  gdk_pixbuf_loader_close (loader, NULL);
416  break;
417  }
418  }
419 
420  tmp_pixbuf = gdk_pixbuf_loader_get_pixbuf (loader);
421  if (tmp_pixbuf) {
422  if ((rc->rotation % 360) == 0)
423  rotated_pixbuf = g_object_ref (tmp_pixbuf);
424  else
425  rotated_pixbuf = gdk_pixbuf_rotate_simple (tmp_pixbuf,
426  360 - rc->rotation);
427  }
428  g_object_unref (loader);
429 
430 out:
431  ev_archive_reset (comics_document->archive);
432  return rotated_pixbuf;
433 }

487 /* Returns a list of file extensions supported by gdk-pixbuf */
488 static GSList*
489 get_supported_image_extensions(void)
490 {
491  GSList *extensions = NULL;
492  GSList *formats = gdk_pixbuf_get_formats ();
493  GSList *l;
494 
495  for (l = formats; l != NULL; l = l->next) {
496  int i;
497  gchar **ext = gdk_pixbuf_format_get_extensions (l->data);
498 
499  for (i = 0; ext[i] != NULL; i++) {
500  extensions = g_slist_append (extensions,
501  g_strdup (ext[i]));
502  }
503 
504  g_strfreev (ext);
505  }
506 
507  g_slist_free (formats);
508  return extensions;
509 }
