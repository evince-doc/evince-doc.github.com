144 static void
145 ev_spawn (const char *uri,
146  GdkScreen *screen,
147  EvLinkDest *dest,
148  EvWindowRunMode mode,
149  const gchar *search_string,
150  guint timestamp)
151 {
152  GString *cmd;
153  gchar *path, *cmdline;
154  GAppInfo *app;
155  GError *error = NULL;
156 
157  cmd = g_string_new (NULL);
158 
159 #ifdef G_OS_WIN32
160 {
161  gchar *dir;
162 
163  dir = g_win32_get_package_installation_directory_of_module (NULL);
164  path = g_build_filename (dir, "bin", "evince", NULL);
165 
166  g_free (dir);
167 }
168 #else
169  path = g_build_filename (BINDIR, "evince", NULL);
170 #endif
171 
172  g_string_append_printf (cmd, " %s", path);
173  g_free (path);
174 
175  /* Page label */
176  if (dest) {
177  switch (ev_link_dest_get_dest_type (dest)) {
178  case EV_LINK_DEST_TYPE_PAGE_LABEL:
179  g_string_append_printf (cmd, " --page-label=%s",
180  ev_link_dest_get_page_label (dest));
181  break;
182  case EV_LINK_DEST_TYPE_PAGE:
183  g_string_append_printf (cmd, " --page-index=%d",
184  ev_link_dest_get_page (dest) + 1);
185  break;
186  case EV_LINK_DEST_TYPE_NAMED:
187  g_string_append_printf (cmd, " --named-dest=%s",
188  ev_link_dest_get_named_dest (dest));
189  break;
190  default:
191  break;
192  }
193  }
194 
195  /* Find string */
196  if (search_string) {
197  g_string_append_printf (cmd, " --find=%s", search_string);
198  }
199 
200  /* Mode */
201  switch (mode) {
202  case EV_WINDOW_MODE_FULLSCREEN:
203  g_string_append (cmd, " -f");
204  break;
205  case EV_WINDOW_MODE_PRESENTATION:
206  g_string_append (cmd, " -s");
207  break;
208  default:
209  break;
210  }
211 
212  cmdline = g_string_free (cmd, FALSE);
213  app = g_app_info_create_from_commandline (cmdline, NULL, 0, &error);
214 
215  if (app != NULL) {
216  GList uri_list;
217  GdkAppLaunchContext *ctx;
218 
219  ctx = gdk_display_get_app_launch_context (gdk_screen_get_display (screen));
220  gdk_app_launch_context_set_screen (ctx, screen);
221  gdk_app_launch_context_set_timestamp (ctx, timestamp);
222 
223  /* Some URIs can be changed when passed through a GFile
224  * (for instance unsupported uris with strange formats like mailto:),
225  * so if you have a textual uri you want to pass in as argument,
226  * consider using g_app_info_launch_uris() instead.
227  * See https://bugzilla.gnome.org/show_bug.cgi?id=644604
228  */
229  uri_list.data = (gchar *)uri;
230  uri_list.prev = uri_list.next = NULL;
231  g_app_info_launch_uris (app, &uri_list,
232  G_APP_LAUNCH_CONTEXT (ctx), &error);
233 
234  g_object_unref (app);
235  g_object_unref (ctx);
236  }
237 
238  if (error != NULL) {
239  g_printerr ("Error launching evince %s: %s\n", uri, error->message);
240  g_error_free (error);
241  }
242 
243  g_free (cmdline);
244 }

322 static void
323 on_register_uri_cb (GObject *source_object,
324  GAsyncResult *res,
325  gpointer user_data)
326 {
327  GDBusConnection *connection = G_DBUS_CONNECTION (source_object);
328  EvRegisterDocData *data = (EvRegisterDocData *)user_data;
329  EvApplication *application = EV_APP;
330  GVariant *value;
331  const gchar *owner;
332  GVariantBuilder builder;
333  GError *error = NULL;
334 
335  g_application_release (G_APPLICATION (application));
336 
337  value = g_dbus_connection_call_finish (connection, res, &error);
338  if (!value) {
339  g_printerr ("Error registering document: %s\n", error->message);
340  g_error_free (error);
341 
342  _ev_application_open_uri_at_dest (application,
343  data->uri,
344  data->screen,
345  data->dest,
346  data->mode,
347  data->search_string,
348  data->timestamp);
349  ev_register_doc_data_free (data);
350 
351  return;
352  }
353 
354  g_variant_get (value, "(&s)", &owner);
355 
356  /* This means that the document wasn't already registered; go
357  * ahead with opening it.
358  */
359  if (owner[0] == '\0') {
360  g_variant_unref (value);
361 
362  application->doc_registered = TRUE;
363 
364  _ev_application_open_uri_at_dest (application,
365  data->uri,
366  data->screen,
367  data->dest,
368  data->mode,
369  data->search_string,
370  data->timestamp);
371  ev_register_doc_data_free (data);
372 
373  return;
374  }
375 
376  /* Already registered */
377  g_variant_builder_init (&builder, G_VARIANT_TYPE ("(a{sv}u)"));
378  g_variant_builder_open (&builder, G_VARIANT_TYPE ("a{sv}"));
379  g_variant_builder_add (&builder, "{sv}",
380  "display",
381  g_variant_new_string (gdk_display_get_name (gdk_screen_get_display (data->screen))));
382  g_variant_builder_add (&builder, "{sv}",
383  "screen",
384  g_variant_new_int32 (gdk_screen_get_number (data->screen)));
385  if (data->dest) {
386  switch (ev_link_dest_get_dest_type (data->dest)) {
387  case EV_LINK_DEST_TYPE_PAGE_LABEL:
388  g_variant_builder_add (&builder, "{sv}", "page-label",
389  g_variant_new_string (ev_link_dest_get_page_label (data->dest)));
390  break;
391  case EV_LINK_DEST_TYPE_PAGE:
392  g_variant_builder_add (&builder, "{sv}", "page-index",
393  g_variant_new_uint32 (ev_link_dest_get_page (data->dest)));
394  break;
395  case EV_LINK_DEST_TYPE_NAMED:
396  g_variant_builder_add (&builder, "{sv}", "named-dest",
397  g_variant_new_string (ev_link_dest_get_named_dest (data->dest)));
398  break;
399  default:
400  break;
401  }
402  }
403  if (data->search_string) {
404  g_variant_builder_add (&builder, "{sv}",
405  "find-string",
406  g_variant_new_string (data->search_string));
407  }
408  if (data->mode != EV_WINDOW_MODE_NORMAL) {
409  g_variant_builder_add (&builder, "{sv}",
410  "mode",
411  g_variant_new_uint32 (data->mode));
412  }
413  g_variant_builder_close (&builder);
414 
415  g_variant_builder_add (&builder, "u", data->timestamp);
416 
417  g_dbus_connection_call (connection,
418  owner,
419  APPLICATION_DBUS_OBJECT_PATH,
420  APPLICATION_DBUS_INTERFACE,
421  "Reload",
422  g_variant_builder_end (&builder),
423  NULL,
424  G_DBUS_CALL_FLAGS_NONE,
425  -1,
426  NULL,
427  on_reload_cb,
428  NULL);
429  g_application_hold (G_APPLICATION (application));
430  g_variant_unref (value);
431  ev_register_doc_data_free (data);
432 }

434 /*
435  * ev_application_register_uri:
436  * @application: The instance of the application.
437  * @uri: The uri to be opened.
438  * @screen: The screen where the link will be shown.
439  * @dest: The #EvLinkDest of the document.
440  * @mode: The run mode of the window.
441  * @search_string: The word or phrase to find in the document.
442  * @timestamp: Current time value.
443  *
444  * Registers @uri with evince-daemon.
445  *
446  */
447 static void
448 ev_application_register_uri (EvApplication *application,
449  const gchar *uri,
450  GdkScreen *screen,
451  EvLinkDest *dest,
452  EvWindowRunMode mode,
453  const gchar *search_string,
454  guint timestamp)
455 {
456  EvRegisterDocData *data;
457 
458  /* If connection hasn't been made fall back to opening without D-BUS features */
459  if (!application->skeleton) {
460  _ev_application_open_uri_at_dest (application, uri, screen, dest, mode, search_string, timestamp);
461  return;
462  }
463 
464  if (application->doc_registered) {
465  /* Already registered, reload */
466  GList *windows, *l;
467 
468  windows = gtk_application_get_windows (GTK_APPLICATION (application));
469  for (l = windows; l != NULL; l = g_list_next (l)) {
470  if (!EV_IS_WINDOW (l->data))
471  continue;
472 
473  ev_application_open_uri_in_window (application, uri,
474  EV_WINDOW (l->data),
475  screen, dest, mode,
476  search_string,
477  timestamp);
478  }
479 
480  return;
481  }
482 
483  data = g_new (EvRegisterDocData, 1);
484  data->uri = g_strdup (uri);
485  data->screen = screen;
486  data->dest = dest ? g_object_ref (dest) : NULL;
487  data->mode = mode;
488  data->search_string = search_string ? g_strdup (search_string) : NULL;
489  data->timestamp = timestamp;
490 
491  g_dbus_connection_call (g_application_get_dbus_connection (G_APPLICATION (application)),
492  EVINCE_DAEMON_SERVICE,
493  EVINCE_DAEMON_OBJECT_PATH,
494  EVINCE_DAEMON_INTERFACE,
495  "RegisterDocument",
496  g_variant_new ("(s)", uri),
497  G_VARIANT_TYPE ("(s)"),
498  G_DBUS_CALL_FLAGS_NONE,
499  -1,
500  NULL,
501  on_register_uri_cb,
502  data);
503 
504  g_application_hold (G_APPLICATION (application));
505 }

713 static gboolean
714 handle_reload_cb (EvEvinceApplication *object,
715  GDBusMethodInvocation *invocation,
716  GVariant *args,
717  guint timestamp,
718  EvApplication *application)
719 {
720  GList *windows, *l;
721  GVariantIter iter;
722  const gchar *key;
723  GVariant *value;
724  GdkDisplay *display = NULL;
725  int screen_number = 0;
726  EvLinkDest *dest = NULL;
727  EvWindowRunMode mode = EV_WINDOW_MODE_NORMAL;
728  const gchar *search_string = NULL;
729  GdkScreen *screen = NULL;
730 
731  g_variant_iter_init (&iter, args);
732 
733  while (g_variant_iter_loop (&iter, "{&sv}", &key, &value)) {
734  if (strcmp (key, "display") == 0 && g_variant_classify (value) == G_VARIANT_CLASS_STRING) {
735  display = ev_display_open_if_needed (g_variant_get_string (value, NULL));
