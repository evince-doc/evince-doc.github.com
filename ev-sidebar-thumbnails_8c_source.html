137 static EvThumbsSizeCache *
138 ev_thumbnails_size_cache_new (EvDocument *document)
139 {
140  EvThumbsSizeCache *cache;
141  gint i, n_pages;
142  EvThumbsSize *thumb_size;
143 
144  cache = g_new0 (EvThumbsSizeCache, 1);
145 
146  if (ev_document_is_page_size_uniform (document)) {
147  cache->uniform = TRUE;
148  get_thumbnail_size_for_page (document, 0,

225 static gboolean
226 ev_sidebar_thumbnails_page_is_in_visible_range (EvSidebarThumbnails *sidebar,
227  guint page)
228 {
229  GtkTreePath *path;
230  GtkTreePath *start, *end;
231  gboolean retval;
232 
233  if (sidebar->priv->tree_view) {
234  GtkTreeSelection *selection;
235  GtkTreeIter iter;
236 
237  selection = gtk_tree_view_get_selection (GTK_TREE_VIEW (sidebar->priv->tree_view));
238  if (!gtk_tree_selection_get_selected (selection, NULL, &iter))
239  return FALSE;
240 
241  path = gtk_tree_model_get_path (GTK_TREE_MODEL (sidebar->priv->list_store), &iter);
242  if (!gtk_tree_view_get_visible_range (GTK_TREE_VIEW (sidebar->priv->tree_view), &start, &end)) {
243  gtk_tree_path_free (path);
244  return FALSE;
245  }
246  } else {
247  GList *selection;
248 
249  selection = gtk_icon_view_get_selected_items (GTK_ICON_VIEW (sidebar->priv->icon_view));
250  if (!selection)
251  return FALSE;
252 
253  path = (GtkTreePath *)selection->data;
254 
255  /* We don't handle or expect multiple selection. */
256  g_assert (selection->next == NULL);
257  g_list_free (selection);
258 
259  if (!gtk_icon_view_get_visible_range (GTK_ICON_VIEW (sidebar->priv->icon_view), &start, &end)) {
260  gtk_tree_path_free (path);
261  return FALSE;
262  }
263  }
264 
265  retval = gtk_tree_path_compare (path, start) >= 0 && gtk_tree_path_compare (path, end) <= 0;
266  gtk_tree_path_free (path);
267  gtk_tree_path_free (start);
268  gtk_tree_path_free (end);
269 
270  return retval;
271 }

418 static void
419 clear_range (EvSidebarThumbnails *sidebar_thumbnails,
420  gint start_page,
421  gint end_page)
422 {
423  EvSidebarThumbnailsPrivate *priv = sidebar_thumbnails->priv;
424  GtkTreePath *path;
425  GtkTreeIter iter;
426  gboolean result;
427  gint prev_width = -1;
428  gint prev_height = -1;
429 
430  g_assert (start_page <= end_page);
431 
432  path = gtk_tree_path_new_from_indices (start_page, -1);
433  for (result = gtk_tree_model_get_iter (GTK_TREE_MODEL (priv->list_store), &iter, path);
434  result && start_page <= end_page;
435  result = gtk_tree_model_iter_next (GTK_TREE_MODEL (priv->list_store), &iter), start_page ++) {
436  EvJobThumbnail *job;
437  cairo_surface_t *loading_icon = NULL;
438  gint width, height;
439 
440  gtk_tree_model_get (GTK_TREE_MODEL (priv->list_store),
441  &iter,
442  COLUMN_JOB, &job,
443  -1);
444 
445  if (job) {
446  g_signal_handlers_disconnect_by_func (job, thumbnail_job_completed_callback, sidebar_thumbnails);
447  ev_job_cancel (EV_JOB (job));
448  g_object_unref (job);
449  }
450 
451  ev_thumbnails_size_cache_get_size (priv->size_cache, start_page,
452  priv->rotation,
453  &width, &height);
454  if (!loading_icon || (width != prev_width && height != prev_height)) {
455  loading_icon =
456  ev_sidebar_thumbnails_get_loading_icon (sidebar_thumbnails,
457  width, height);
458  }
459 
460  prev_width = width;
461  prev_height = height;
462 
463  gtk_list_store_set (priv->list_store, &iter,
464  COLUMN_JOB, NULL,
465  COLUMN_THUMBNAIL_SET, FALSE,
466  COLUMN_SURFACE, loading_icon,
467  -1);
468  }
469  gtk_tree_path_free (path);
470 }

498 static void
499 add_range (EvSidebarThumbnails *sidebar_thumbnails,
500  gint start_page,
501  gint end_page)
502 {
503  EvSidebarThumbnailsPrivate *priv = sidebar_thumbnails->priv;
504  GtkTreePath *path;
505  GtkTreeIter iter;
506  gboolean result;
507  gint page = start_page;
508 
509  g_assert (start_page <= end_page);
510 
511  path = gtk_tree_path_new_from_indices (start_page, -1);
512  for (result = gtk_tree_model_get_iter (GTK_TREE_MODEL (priv->list_store), &iter, path);
513  result && page <= end_page;
514  result = gtk_tree_model_iter_next (GTK_TREE_MODEL (priv->list_store), &iter), page ++) {
515  EvJob *job;
516  gboolean thumbnail_set;
517 
518  gtk_tree_model_get (GTK_TREE_MODEL (priv->list_store), &iter,
519  COLUMN_JOB, &job,
520  COLUMN_THUMBNAIL_SET, &thumbnail_set,
521  -1);
522 
523  if (job == NULL && !thumbnail_set) {
524  gint thumbnail_width, thumbnail_height;
525  get_size_for_page (sidebar_thumbnails, page, &thumbnail_width, &thumbnail_height);
526 
527  job = ev_job_thumbnail_new_with_target_size (priv->document,
528  page, priv->rotation,
529  thumbnail_width, thumbnail_height);
530  ev_job_thumbnail_set_has_frame (EV_JOB_THUMBNAIL (job), FALSE);
531  ev_job_thumbnail_set_output_format (EV_JOB_THUMBNAIL (job), EV_JOB_THUMBNAIL_SURFACE);
532  g_object_set_data_full (G_OBJECT (job), "tree_iter",
533  gtk_tree_iter_copy (&iter),
534  (GDestroyNotify) gtk_tree_iter_free);
535  g_signal_connect (job, "finished",
536  G_CALLBACK (thumbnail_job_completed_callback),
537  sidebar_thumbnails);
538  gtk_list_store_set (priv->list_store, &iter,
539  COLUMN_JOB, job,
540  -1);
541  ev_job_scheduler_push_job (EV_JOB (job), EV_JOB_PRIORITY_HIGH);
542 
543  /* The queue and the list own a ref to the job now */
544  g_object_unref (job);
545  } else if (job) {
546  g_object_unref (job);
547  }
548  }
549  gtk_tree_path_free (path);
550 }

552 /* This modifies start */
553 static void
554 update_visible_range (EvSidebarThumbnails *sidebar_thumbnails,
555  gint start_page,
556  gint end_page)
557 {
558  EvSidebarThumbnailsPrivate *priv = sidebar_thumbnails->priv;
559  int old_start_page, old_end_page;
560 
561  old_start_page = priv->start_page;
562  old_end_page = priv->end_page;
563 
564  if (start_page == old_start_page &&
565  end_page == old_end_page)
566  return;
567 
568  /* Clear the areas we no longer display */
569  if (old_start_page >= 0 && old_start_page < start_page)
570  clear_range (sidebar_thumbnails, old_start_page, MIN (start_page - 1, old_end_page));
571 
572  if (old_end_page > 0 && old_end_page > end_page)
573  clear_range (sidebar_thumbnails, MAX (end_page + 1, old_start_page), old_end_page);
574 
575  add_range (sidebar_thumbnails, start_page, end_page);
576 
577  priv->start_page = start_page;
578  priv->end_page = end_page;
579 }

581 static void
582 adjustment_changed_cb (EvSidebarThumbnails *sidebar_thumbnails)
583 {
584  EvSidebarThumbnailsPrivate *priv = sidebar_thumbnails->priv;
585  GtkTreePath *path = NULL;
586  GtkTreePath *path2 = NULL;
587  gdouble page_size;
588  gdouble value;
589  gint wy1;
590  gint wy2;
591 
592  /* Widget is not currently visible */
593  if (!gtk_widget_get_mapped (GTK_WIDGET (sidebar_thumbnails)))
594  return;
595 
596  page_size = gtk_adjustment_get_page_size (priv->vadjustment);
597 
598  if (page_size == 0)
599  return;
600 
601  value = gtk_adjustment_get_value (priv->vadjustment);
602 
603  if (priv->tree_view) {
604  if (! gtk_widget_get_realized (priv->tree_view))
605  return;
606 
607  gtk_tree_view_convert_tree_to_bin_window_coords (GTK_TREE_VIEW (priv->tree_view),
608  0, (int) value,
609  NULL, &wy1);
610  gtk_tree_view_convert_tree_to_bin_window_coords (GTK_TREE_VIEW (priv->tree_view),
611  0, (int) (value + page_size),
612  NULL, &wy2);
613  gtk_tree_view_get_path_at_pos (GTK_TREE_VIEW (priv->tree_view),
614  1, wy1 + 1, &path,
615  NULL, NULL, NULL);
616  gtk_tree_view_get_path_at_pos (GTK_TREE_VIEW (priv->tree_view),
617  1, wy2 -1, &path2,
618  NULL, NULL, NULL);
619  } else if (priv->icon_view) {
620  if (! gtk_widget_get_realized (priv->icon_view))
621  return;
622  if (! gtk_icon_view_get_visible_range (GTK_ICON_VIEW (priv->icon_view), &path, &path2))
623  return;
624  } else {
625  return;
626  }
627 
628  if (path && path2) {
629  update_visible_range (sidebar_thumbnails,
630  gtk_tree_path_get_indices (path)[0],
631  gtk_tree_path_get_indices (path2)[0]);
632  }
633 
634  gtk_tree_path_free (path);
635  gtk_tree_path_free (path2);
636 }

967 static void
968 ev_sidebar_thumbnails_document_changed_cb (EvDocumentModel *model,
969  GParamSpec *pspec,
970  EvSidebarThumbnails *sidebar_thumbnails)
971 {
972  EvDocument *document = ev_document_model_get_document (model);
973  EvSidebarThumbnailsPrivate *priv = sidebar_thumbnails->priv;
974 
975  if (ev_document_get_n_pages (document) <= 0 ||
976  !ev_document_check_dimensions (document)) {
977  return;
978  }
979 
980  priv->size_cache = ev_thumbnails_size_cache_get (document);
981  priv->document = document;
982  priv->n_pages = ev_document_get_n_pages (document);
983  priv->rotation = ev_document_model_get_rotation (model);
984  priv->inverted_colors = ev_document_model_get_inverted_colors (model);
985  priv->loading_icons = g_hash_table_new_full (g_str_hash,
986  g_str_equal,
987  (GDestroyNotify)g_free,
988  (GDestroyNotify)cairo_surface_destroy);
989 
990  ev_sidebar_thumbnails_clear_model (sidebar_thumbnails);
991  ev_sidebar_thumbnails_fill_model (sidebar_thumbnails);
992 
993  /* Create the view widget, and remove the old one, if needed */
994  if (ev_sidebar_thumbnails_use_icon_view (sidebar_thumbnails)) {
995  if (priv->tree_view) {
996  gtk_container_remove (GTK_CONTAINER (priv->swindow), priv->tree_view);
997  priv->tree_view = NULL;
998  }
999 
1000  if (! priv->icon_view) {
1001  ev_sidebar_init_icon_view (sidebar_thumbnails);
1002  g_object_notify (G_OBJECT (sidebar_thumbnails), "main_widget");
1003  } else {
1004  gtk_widget_queue_resize (priv->icon_view);
1005  }
1006  } else {
1007  if (priv->icon_view) {
1008  gtk_container_remove (GTK_CONTAINER (priv->swindow), priv->icon_view);
1009  priv->icon_view = NULL;
1010  }
1011 
1012  if (! priv->tree_view) {
1013  ev_sidebar_init_tree_view (sidebar_thumbnails);
1014  g_object_notify (G_OBJECT (sidebar_thumbnails), "main_widget");
1015  }
1016  }
1017 
1018  /* Connect to the signal and trigger a fake callback */
1019  g_signal_connect_swapped (priv->model, "page-changed",
1020  G_CALLBACK (page_changed_cb),
1021  sidebar_thumbnails);
1022  g_signal_connect (priv->model, "notify::rotation",
1023  G_CALLBACK (ev_sidebar_thumbnails_rotation_changed_cb),
1024  sidebar_thumbnails);
1025  g_signal_connect (priv->model, "notify::inverted-colors",
1026  G_CALLBACK (ev_sidebar_thumbnails_inverted_colors_changed_cb),
1027  sidebar_thumbnails);
1028  sidebar_thumbnails->priv->start_page = -1;
1029  sidebar_thumbnails->priv->end_page = -1;
1030  ev_sidebar_thumbnails_set_current_page (sidebar_thumbnails,
1031  ev_document_model_get_page (model));
1032  adjustment_changed_cb (sidebar_thumbnails);
1033 }
