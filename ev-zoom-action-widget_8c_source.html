72 static void
73 ev_zoom_action_widget_set_zoom_level (EvZoomActionWidget *control,
74  float zoom)
75 {
76  GtkWidget *entry = gtk_bin_get_child (GTK_BIN (control->priv->combo));
77  gchar *zoom_str;
78  float zoom_perc;
79  guint i;
80 
81  for (i = 3; i < G_N_ELEMENTS (zoom_levels); i++) {
82  if (ABS (zoom - zoom_levels[i].level) < EPSILON) {
83  gtk_entry_set_text (GTK_ENTRY (entry), zoom_levels[i].name);
84  return;
85  }
86  }
87 
88  zoom_perc = zoom * 100.;
89  if (ABS ((gint)zoom_perc - zoom_perc) < 0.001)
90  zoom_str = g_strdup_printf ("%d%%", (gint)zoom_perc);
91  else
92  zoom_str = g_strdup_printf ("%.2f%%", zoom_perc);
93  gtk_entry_set_text (GTK_ENTRY (entry), zoom_str);
94  g_free (zoom_str);
95 }

99 {

126 static void
127 combo_changed_cb (GtkComboBox *combo,
128  EvZoomActionWidget *control)
129 {
130  gint index;
131  float zoom;
132  EvSizingMode mode;
133 
134  index = gtk_combo_box_get_active (combo);
135  if (index == -1)
136  return;
137 
138  zoom = zoom_levels[index].level;
139  if (zoom == EV_ZOOM_FIT_PAGE)
140  mode = EV_SIZING_FIT_PAGE;
141  else if (zoom == EV_ZOOM_FIT_WIDTH)
142  mode = EV_SIZING_FIT_WIDTH;
143  else
144  mode = EV_SIZING_FREE;
145 
146  ev_document_model_set_sizing_mode (control->priv->model, mode);
147  if (mode == EV_SIZING_FREE) {
148  GdkScreen *screen = gtk_widget_get_screen (GTK_WIDGET (control));
149 
150  ev_document_model_set_scale (control->priv->model,
151  zoom * ev_document_misc_get_screen_dpi (screen) / 72.0);
152  }
153 }

155 static void
156 combo_activated_cb (GtkEntry *entry,
157  EvZoomActionWidget *control)
158 {
159  GdkScreen *screen;
160  double zoom_perc;
161  float zoom;
162  const gchar *text = gtk_entry_get_text (entry);
163  gchar *end_ptr = NULL;
164 
165  if (!text || text[0] == '\0') {
166  ev_zoom_action_widget_update_zoom_level (control);
167  return;
168  }
169 
170  zoom_perc = g_strtod (text, &end_ptr);
171  if (end_ptr && end_ptr[0] != '\0' && end_ptr[0] != '%') {
172  ev_zoom_action_widget_update_zoom_level (control);
173  return;
174  }
175 
176  screen = gtk_widget_get_screen (GTK_WIDGET (control));
177  zoom = zoom_perc / 100.;
178  ev_document_model_set_sizing_mode (control->priv->model, EV_SIZING_FREE);
179  ev_document_model_set_scale (control->priv->model,
180  zoom * ev_document_misc_get_screen_dpi (screen) / 72.0);
181 }

258 ev_zoom_action_widget_init (EvZoomActionWidget *control)
259 {
260  EvZoomActionWidgetPrivate *priv;
261  GtkWidget *entry;
262  GtkWidget *vbox;
263  GtkListStore *store;
264 
265  control->priv = G_TYPE_INSTANCE_GET_PRIVATE (control, EV_TYPE_ZOOM_ACTION_WIDGET, EvZoomActionWidgetPrivate);
266  priv = control->priv;
267 
268  store = gtk_list_store_new (N_COLUMNS, G_TYPE_STRING, G_TYPE_BOOLEAN);
269  fill_combo_model (store, G_N_ELEMENTS (zoom_levels));
270 
271  priv->combo = gtk_combo_box_new_with_model_and_entry (GTK_TREE_MODEL (store));
272  gtk_combo_box_set_focus_on_click (GTK_COMBO_BOX (priv->combo), FALSE);
273  gtk_combo_box_set_entry_text_column (GTK_COMBO_BOX (priv->combo), TEXT_COLUMN);
274  gtk_combo_box_set_row_separator_func (GTK_COMBO_BOX (priv->combo),
275  (GtkTreeViewRowSeparatorFunc)row_is_separator,
276  NULL, NULL);
277  g_object_unref (store);
278 
279  entry = gtk_bin_get_child (GTK_BIN (priv->combo));
280  gtk_entry_set_width_chars (GTK_ENTRY (entry), get_max_zoom_level_label ());
281 
282  vbox = gtk_box_new (GTK_ORIENTATION_VERTICAL, 0);
283  gtk_box_pack_start (GTK_BOX (vbox), priv->combo, TRUE, FALSE, 0);
284  gtk_widget_show (priv->combo);
285 
286  gtk_container_add (GTK_CONTAINER (control), vbox);
287  gtk_widget_show (vbox);
288 
289  priv->combo_changed_handler =

314 void
315 ev_zoom_action_widget_set_model (EvZoomActionWidget *control,
316  EvDocumentModel *model)
317 {
318  g_return_if_fail (EV_IS_ZOOM_ACTION_WIDGET (control));
319 
320  if (control->priv->model == model)
321  return;
322 
323  if (control->priv->model) {
324  g_object_remove_weak_pointer (G_OBJECT (control->priv->model),
325  (gpointer)&control->priv->model);
326  g_signal_handlers_disconnect_by_func (control->priv->model,
327  G_CALLBACK (zoom_changed_cb),
328  control);
329  }
330  control->priv->model = model;
331  if (!model)
332  return;
333 
334  g_object_add_weak_pointer (G_OBJECT (model),
335  (gpointer)&control->priv->model);
336 
337  if (ev_document_model_get_document (model)) {
338  ev_zoom_action_widget_update_zoom_level (control);
339  } else {
340  ev_zoom_action_widget_set_zoom_level (control, 1.);
341  }
342 
343  g_signal_connect (control->priv->model, "notify::document",
344  G_CALLBACK (document_changed_cb),
345  control);
346  g_signal_connect (control->priv->model, "notify::scale",
347  G_CALLBACK (zoom_changed_cb),
348  control);
349 }

351 void
352 ev_zoom_action_widget_set_max_zoom_level (EvZoomActionWidget *control,
353  float max_zoom)
354 {
355  EvZoomActionWidgetPrivate *priv;
356  GtkListStore *model;
357  guint max_level_index = 3;
358  guint i;
359 
360  g_return_if_fail (EV_IS_ZOOM_ACTION_WIDGET (control));
361 
362  priv = control->priv;
363 
364  for (i = 3; i < G_N_ELEMENTS (zoom_levels); i++, max_level_index++) {
365  if (zoom_levels[i].level > max_zoom)
366  break;
367  }
368 
369  g_signal_handler_block (priv->combo, priv->combo_changed_handler);
370 
371  model = GTK_LIST_STORE (gtk_combo_box_get_model (GTK_COMBO_BOX (priv->combo)));
372  gtk_list_store_clear (model);
373  fill_combo_model (model, max_level_index);
374 
375  g_signal_handler_unblock (priv->combo, priv->combo_changed_handler);
376 }
