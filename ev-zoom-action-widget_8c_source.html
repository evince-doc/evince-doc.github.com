46 static void
47 ev_zoom_action_widget_set_zoom_level (EvZoomActionWidget *control,
48  float zoom)
49 {
50  gchar *zoom_str;
51  float zoom_perc;
52  guint i;
53 
54  for (i = 3; i < G_N_ELEMENTS (zoom_levels); i++) {
55  if (ABS (zoom - zoom_levels[i].level) < EPSILON) {
56  gtk_entry_set_text (GTK_ENTRY (control->priv->entry),
57  zoom_levels[i].name);
58  return;
59  }
60  }
61 
62  zoom_perc = zoom * 100.;
63  if (ABS ((gint)zoom_perc - zoom_perc) < 0.001)
64  zoom_str = g_strdup_printf ("%d%%", (gint)zoom_perc);
65  else
66  zoom_str = g_strdup_printf ("%.2f%%", zoom_perc);
67  gtk_entry_set_text (GTK_ENTRY (control->priv->entry), zoom_str);
68  g_free (zoom_str);
69 }

100 static void
101 entry_activated_cb (GtkEntry *entry,
102  EvZoomActionWidget *control)
103 {
104  GdkScreen *screen;
105  double zoom_perc;
106  float zoom;
107  const gchar *text = gtk_entry_get_text (entry);
108  gchar *end_ptr = NULL;
109 
110  if (!text || text[0] == '\0') {
111  ev_zoom_action_widget_update_zoom_level (control);
112  return;
113  }
114 
115  zoom_perc = g_strtod (text, &end_ptr);
116  if (end_ptr && end_ptr[0] != '\0' && end_ptr[0] != '%') {
117  ev_zoom_action_widget_update_zoom_level (control);
118  return;
119  }
120 
121  screen = gtk_widget_get_screen (GTK_WIDGET (control));
122  zoom = zoom_perc / 100.;
123  ev_document_model_set_sizing_mode (control->priv->model, EV_SIZING_FREE);
124  ev_document_model_set_scale (control->priv->model,
125  zoom * ev_document_misc_get_screen_dpi (screen) / 72.0);
126 }

136 static gint
137 get_max_zoom_level_label (EvZoomActionWidget *control)
138 {
139  GList *actions, *l;
140  gint width = 0;
141 
142  actions = gtk_action_group_list_actions (ev_window_get_zoom_selector_action_group (control->priv->window));
143 
144  for (l = actions; l; l = g_list_next (l)) {
145  GtkAction *action = (GtkAction *)l->data;
146  int length;
147 
148  length = g_utf8_strlen (gtk_action_get_label (action), -1);
149  if (length > width)
150  width = length;
151  }
152 
153  g_list_free (actions);
154 
155  /* Count the toggle size as one character more */
156  return width + 1;
157 }

219 static void
220 menu_position_below (GtkMenu *menu,
221  gint *x,
222  gint *y,
223  gint *push_in,
224  gpointer user_data)
225 {
226  EvZoomActionWidget *control;
227  GtkWidget *widget;
228  GtkAllocation child_allocation;
229  GtkRequisition req;
230  GdkScreen *screen;
231  gint monitor_num;
232  GdkRectangle monitor;
233  gint sx = 0, sy = 0;
234 
235  control = EV_ZOOM_ACTION_WIDGET (user_data);
236  widget = GTK_WIDGET (control);
237 
238  gtk_widget_get_allocation (control->priv->entry, &child_allocation);
239 
240  if (!gtk_widget_get_has_window (control->priv->entry)) {
241  sx += child_allocation.x;
242  sy += child_allocation.y;
243  }
244 
245  gdk_window_get_root_coords (gtk_widget_get_window (control->priv->entry),
246  sx, sy, &sx, &sy);
247 
248  gtk_widget_get_preferred_size (GTK_WIDGET (menu), &req, NULL);
249 
250  if (gtk_widget_get_direction (widget) == GTK_TEXT_DIR_LTR)
251  *x = sx;
252  else
253  *x = sx + child_allocation.width - req.width;
254  *y = sy;
255 
256  screen = gtk_widget_get_screen (widget);
257  monitor_num = gdk_screen_get_monitor_at_window (screen,
258  gtk_widget_get_window (widget));
259  gdk_screen_get_monitor_workarea (screen, monitor_num, &monitor);
260 
261  if (*x < monitor.x)
262  *x = monitor.x;
263  else if (*x + req.width > monitor.x + monitor.width)
264  *x = monitor.x + monitor.width - req.width;
265 
266  if (monitor.y + monitor.height - *y - child_allocation.height >= req.height)
267  *y += child_allocation.height;
268  else if (*y - monitor.y >= req.height)
269  *y -= req.height;
270  else if (monitor.y + monitor.height - *y - child_allocation.height > *y - monitor.y)
271  *y += child_allocation.height;
272  else
273  *y -= req.height;
274 
275  *push_in = FALSE;
276 }

376 void
377 ev_zoom_action_widget_set_model (EvZoomActionWidget *control,
378  EvDocumentModel *model)
379 {
380  g_return_if_fail (EV_IS_ZOOM_ACTION_WIDGET (control));
381 
382  if (control->priv->model == model)
383  return;
384 
385  if (control->priv->model) {
386  g_object_remove_weak_pointer (G_OBJECT (control->priv->model),
387  (gpointer)&control->priv->model);
388  g_signal_handlers_disconnect_by_func (control->priv->model,
389  G_CALLBACK (zoom_changed_cb),
390  control);
391  }
392  control->priv->model = model;
393  if (!model)
394  return;
395 
396  g_object_add_weak_pointer (G_OBJECT (model),
397  (gpointer)&control->priv->model);
398 
399  if (ev_document_model_get_document (model)) {
400  ev_zoom_action_widget_update_zoom_level (control);
401  } else {
402  ev_zoom_action_widget_set_zoom_level (control, 1.);
403  }
404 
405  g_signal_connect (control->priv->model, "notify::document",
406  G_CALLBACK (document_changed_cb),
407  control);
408  g_signal_connect (control->priv->model, "notify::scale",
409  G_CALLBACK (zoom_changed_cb),
410  control);
411 }
