248 /* convenience macro to ease interface addition in the CODE
249  * section of EV_BACKEND_REGISTER_WITH_CODE (this macro relies on
250  * the g_define_type_id present within EV_BACKEND_REGISTER_WITH_CODE()).
251  * usage example:
252  * EV_BACKEND_REGISTER_WITH_CODE (PdfDocument, pdf_document,
253  * EV_BACKEND_IMPLEMENT_INTERFACE (EV_TYPE_DOCUMENT_THUMBNAILS,
254  * pdf_document_document_thumbnails_iface_init));
255  */
256 #define EV_BACKEND_IMPLEMENT_INTERFACE(TYPE_IFACE, iface_init) { \
257  const GInterfaceInfo g_implement_interface_info = { \
258  (GInterfaceInitFunc) iface_init, NULL, NULL \
259  }; \
260  g_type_module_add_interface (module, \
261  g_define_type_id, \
262  TYPE_IFACE, \
263  &g_implement_interface_info); \
264 }

266 /*
267  * Utility macro used to register backends
268  *
269  * use: EV_BACKEND_REGISTER_WITH_CODE(BackendName, backend_name, CODE)
270  */
271 #define EV_BACKEND_REGISTER_WITH_CODE(BackendName, backend_name, CODE) \
272  \
273 static GType g_define_type_id = 0; \
274  \
275 GType \
276 backend_name##_get_type (void) \
277 { \
278  return g_define_type_id; \
279 } \
280  \
281 static void backend_name##_init (BackendName *self); \
282 static void backend_name##_class_init (BackendName##Class *klass); \
283 static gpointer backend_name##_parent_class = NULL; \
284 static void backend_name##_class_intern_init (gpointer klass) \
285 { \
286  backend_name##_parent_class = g_type_class_peek_parent (klass); \
287  backend_name##_class_init ((BackendName##Class *) klass); \
288 } \
289  \
290 G_MODULE_EXPORT GType \
291 register_evince_backend (GTypeModule *module) \
292 { \
293  const GTypeInfo our_info = { \
294  sizeof (BackendName##Class), \
295  NULL, /* base_init */ \
296  NULL, /* base_finalize */ \
297  (GClassInitFunc) backend_name##_class_intern_init, \
298  NULL, \
299  NULL, /* class_data */ \
300  sizeof (BackendName), \
301  0, /* n_preallocs */ \
302  (GInstanceInitFunc) backend_name##_init \
303  }; \
304  \
305  /* Initialise the i18n stuff */ \
306  bindtextdomain (GETTEXT_PACKAGE, GNOMELOCALEDIR); \
307  bind_textdomain_codeset (GETTEXT_PACKAGE, "UTF-8"); \
308  \
309  g_define_type_id = g_type_module_register_type (module, \
310  EV_TYPE_DOCUMENT, \
311  #BackendName, \
312  &our_info, \
313  (GTypeFlags)0); \
314  \
315  CODE \
316  \
317  return g_define_type_id; \
318 }
