207 static gboolean
208 sm_client_xsmp_set_initial_properties (gpointer user_data)
209 {
210  EggSMClientXSMP *xsmp = user_data;

212  GPtrArray *clone, *restart;
213  char pid_str[64];
214 
215  if (xsmp->idle)

217  g_source_remove (xsmp->idle);
218  xsmp->idle = 0;
219  }
220  xsmp->waiting_to_set_initial_properties = FALSE;
221 
222  if (egg_sm_client_get_mode () == EGG_SM_CLIENT_MODE_NO_RESTART)
223  xsmp->restart_style = SmRestartNever;
224 
225  /* Parse info out of desktop file */
226  desktop_file = egg_get_desktop_file ();
227  if (desktop_file)
228  {
229  GError *err = NULL;

232

251  g_warning ("Could not parse Exec line in desktop file: %s",

254  }
255  g_free (cmdline);
256  }
257  }
258

269  * useful with it.
270  */

272  set_properties (xsmp,

278  card8_prop (SmRestartStyleHint, xsmp->restart_style),
279  NULL);
280  g_ptr_array_free (clone, TRUE);
281  g_ptr_array_free (restart, TRUE);
282 
283  if (desktop_file)
284  {
285  set_properties (xsmp,
286  string_prop ("_GSM_DesktopFile", egg_desktop_file_get_source (desktop_file)),
287  NULL);
288  }
289 
290  update_pending_events (xsmp);
291  return FALSE;

317 static void
318 sm_client_xsmp_startup (EggSMClient *client,
319  const char *client_id)
320 {
321  EggSMClientXSMP *xsmp = (EggSMClientXSMP *)client;
322  SmcCallbacks callbacks;
323  char *ret_client_id;
324  char error_string_ret[256];
325 
326  xsmp->client_id = g_strdup (client_id);
327 
328  ice_init ();
329  SmcSetErrorHandler (smc_error_handler);
330 
331  callbacks.save_yourself.callback = xsmp_save_yourself;
332  callbacks.die.callback = xsmp_die;
333  callbacks.save_complete.callback = xsmp_save_complete;
334  callbacks.shutdown_cancelled.callback = xsmp_shutdown_cancelled;
335 
336  callbacks.save_yourself.client_data = xsmp;
337  callbacks.die.client_data = xsmp;
338  callbacks.save_complete.client_data = xsmp;
339  callbacks.shutdown_cancelled.client_data = xsmp;
340 
341  client_id = NULL;
342  error_string_ret[0] = '\0';
343  xsmp->connection =
344  SmcOpenConnection (NULL, xsmp, SmProtoMajor, SmProtoMinor,
345  SmcSaveYourselfProcMask | SmcDieProcMask |
346  SmcSaveCompleteProcMask |
347  SmcShutdownCancelledProcMask,
348  &callbacks,
349  xsmp->client_id, &ret_client_id,
350  sizeof (error_string_ret), error_string_ret);
351 
352  if (!xsmp->connection)
353  {
354  g_warning ("Failed to connect to the session manager: %s\n",
355  error_string_ret[0] ?
356  error_string_ret : "no error message given");
357  xsmp->state = XSMP_STATE_CONNECTION_CLOSED;
358  return;
359  }
360 
361  /* We expect a pointless initial SaveYourself if either (a) we
362  * didn't have an initial client ID, or (b) we DID have an initial
363  * client ID, but the server rejected it and gave us a new one.
364  */
365  if (!xsmp->client_id ||
366  (ret_client_id && strcmp (xsmp->client_id, ret_client_id) != 0))
367  xsmp->expecting_initial_save_yourself = TRUE;
368 
369  if (ret_client_id)
370  {
371  g_free (xsmp->client_id);
372  xsmp->client_id = g_strdup (ret_client_id);
373  free (ret_client_id);
374 
375  gdk_threads_enter ();
376 #if !GTK_CHECK_VERSION(2,23,3) && !GTK_CHECK_VERSION(3,0,0)
377  gdk_set_sm_client_id (xsmp->client_id);
378 #else
379  gdk_x11_set_sm_client_id (xsmp->client_id);
380 #endif
381  gdk_threads_leave ();
382 
383  g_debug ("Got client ID \"%s\"", xsmp->client_id);
384  }
385 
386  xsmp->state = XSMP_STATE_IDLE;
387 
388  /* Do not set the initial properties until we reach the main loop,
389  * so that the application has a chance to call
390  * egg_set_desktop_file(). (This may also help the session manager
391  * have a better idea of when the application is fully up and
392  * running.)
393  */
394  xsmp->waiting_to_set_initial_properties = TRUE;
395  xsmp->idle = g_idle_add (sm_client_xsmp_set_initial_properties, client);
396 }

416 static void
417 sm_client_xsmp_will_quit (EggSMClient *client,
418  gboolean will_quit)
419 {
420  EggSMClientXSMP *xsmp = (EggSMClientXSMP *)client;
421 
422  if (xsmp->state == XSMP_STATE_CONNECTION_CLOSED)
423  {
424  /* The session manager has already exited! Schedule a quit
425  * signal.
426  */
427  xsmp->waiting_to_emit_quit = TRUE;
428  update_pending_events (xsmp);
429  return;
430  }
431  else if (xsmp->state == XSMP_STATE_SHUTDOWN_CANCELLED)
432  {
433  /* We received a ShutdownCancelled message while the application
434  * was interacting; Schedule a quit_cancelled signal.
435  */
436  xsmp->waiting_to_emit_quit_cancelled = TRUE;
437  update_pending_events (xsmp);
438  return;
439  }
440 
441  g_return_if_fail (xsmp->state == XSMP_STATE_INTERACT);
442 
443  g_debug ("Sending InteractDone(%s)", will_quit ? "False" : "True");
444  SmcInteractDone (xsmp->connection, !will_quit);
445 
446  if (will_quit && xsmp->need_save_state)
447  save_state (xsmp);
448 
449  g_debug ("Sending SaveYourselfDone(%s)", will_quit ? "True" : "False");
450  SmcSaveYourselfDone (xsmp->connection, will_quit);
451  xsmp->state = XSMP_STATE_SAVE_YOURSELF_DONE;
452 }

454 static gboolean
455 sm_client_xsmp_end_session (EggSMClient *client,
456  EggSMClientEndStyle style,
457  gboolean request_confirmation)
458 {
459  EggSMClientXSMP *xsmp = (EggSMClientXSMP *)client;
460  int save_type;
461 
462  /* To end the session via XSMP, we have to send a
463  * SaveYourselfRequest. We aren't allowed to do that if anything
464  * else is going on, but we don't want to expose this fact to the
465  * application. So we do our best to patch things up here...
466  *
467  * In the worst case, this method might block for some length of
468  * time in process_ice_messages, but the only time that code path is
469  * honestly likely to get hit is if the application tries to end the
470  * session as the very first thing it does, in which case it
471  * probably won't actually block anyway. It's not worth gunking up
472  * the API to try to deal nicely with the other 0.01% of cases where
473  * this happens.
474  */
475 
476  while (xsmp->state != XSMP_STATE_IDLE ||
477  xsmp->expecting_initial_save_yourself)
478  {
479  /* If we're already shutting down, we don't need to do anything. */
480  if (xsmp->shutting_down)
481  return TRUE;
482 
483  switch (xsmp->state)
484  {
485  case XSMP_STATE_CONNECTION_CLOSED:
486  return FALSE;
487 
488  case XSMP_STATE_SAVE_YOURSELF:
489  /* Trying to log out from the save_state callback? Whatever.
490  * Abort the save_state.
491  */
492  SmcSaveYourselfDone (xsmp->connection, FALSE);
493  xsmp->state = XSMP_STATE_SAVE_YOURSELF_DONE;
494  break;
495 
496  case XSMP_STATE_INTERACT_REQUEST:
497  case XSMP_STATE_INTERACT:
498  case XSMP_STATE_SHUTDOWN_CANCELLED:
499  /* Already in a shutdown-related state, just ignore
500  * the new shutdown request...
501  */
502  return TRUE;
503 
504  case XSMP_STATE_IDLE:
505  if (xsmp->waiting_to_set_initial_properties)
506  sm_client_xsmp_set_initial_properties (xsmp);
507 
508  if (!xsmp->expecting_initial_save_yourself)
509  break;
510  /* else fall through */
511 
512  case XSMP_STATE_SAVE_YOURSELF_DONE:
513  /* We need to wait for some response from the server.*/
514  process_ice_messages (SmcGetIceConnection (xsmp->connection));
515  break;
516 
517  default:
518  /* Hm... shouldn't happen */
519  return FALSE;
520  }
521  }
522 
523  /* xfce4-session will do the wrong thing if we pass SmSaveGlobal and
524  * the user chooses to save the session. But gnome-session will do
525  * the wrong thing if we pass SmSaveBoth and the user chooses NOT to
526  * save the session... Sigh.
527  */
528  if (!strcmp (SmcVendor (xsmp->connection), "xfce4-session"))
529  save_type = SmSaveBoth;
530  else
531  save_type = SmSaveGlobal;
532 
533  g_debug ("Sending SaveYourselfRequest(SmSaveGlobal, Shutdown, SmInteractStyleAny, %sFast)", request_confirmation ? "!" : "");
534  SmcRequestSaveYourself (xsmp->connection,
535  save_type,
536  True, /* shutdown */
537  SmInteractStyleAny,
538  !request_confirmation, /* fast */
539  True /* global */);
540  return TRUE;
541 }

543 static gboolean
544 idle_do_pending_events (gpointer data)
545 {
546  EggSMClientXSMP *xsmp = data;
547  EggSMClient *client = data;
548 
549  gdk_threads_enter ();
550 
551  xsmp->idle = 0;
552 
553  if (xsmp->waiting_to_emit_quit)
554  {
555  xsmp->waiting_to_emit_quit = FALSE;
556  egg_sm_client_quit (client);
557  goto out;
558  }
559 
560  if (xsmp->waiting_to_emit_quit_cancelled)
561  {
562  xsmp->waiting_to_emit_quit_cancelled = FALSE;
563  egg_sm_client_quit_cancelled (client);
564  xsmp->state = XSMP_STATE_IDLE;
565  }
566 
567  if (xsmp->waiting_to_save_myself)
568  {
569  xsmp->waiting_to_save_myself = FALSE;
570  do_save_yourself (xsmp);
571  }
572 
573  out:
574  gdk_threads_leave ();
575  return FALSE;
576 }

578 static void
579 update_pending_events (EggSMClientXSMP *xsmp)
580 {
581  gboolean want_idle =
582  xsmp->waiting_to_emit_quit ||
583  xsmp->waiting_to_emit_quit_cancelled ||
584  xsmp->waiting_to_save_myself;
585 
586  if (want_idle)
587  {
588  if (xsmp->idle == 0)
589  xsmp->idle = g_idle_add (idle_do_pending_events, xsmp);
590  }
591  else
592  {
593  if (xsmp->idle != 0)
594  g_source_remove (xsmp->idle);
595  xsmp->idle = 0;
596  }

600 fix_broken_state (EggSMClientXSMP *xsmp, const char *message,
601  gboolean send_interact_done,
602  gboolean send_save_yourself_done)
603 {
604  g_warning ("Received XSMP %s message in state %s: client or server error",

606 
607  /* Forget any pending SaveYourself plans we had */
608  xsmp->waiting_to_save_myself = FALSE;

622 xsmp_save_yourself (SmcConn smc_conn,
623  SmPointer client_data,
624  int save_type,
625  Bool shutdown,
626  int interact_style,
627  Bool fast)
628 {
629  EggSMClientXSMP *xsmp = client_data;
630  gboolean wants_quit_requested;

637  interact_style == SmInteractStyleErrors ? "SmInteractStyleErrors" :
638  "SmInteractStyleNone", fast ? "Fast" : "!Fast",
639  EGG_SM_CLIENT_XSMP_STATE (xsmp));
640 
641  if (xsmp->state != XSMP_STATE_IDLE &&
642  xsmp->state != XSMP_STATE_SHUTDOWN_CANCELLED)
643  {
644  fix_broken_state (xsmp, "SaveYourself", FALSE, TRUE);
645  return;
646  }
647 
648  if (xsmp->waiting_to_set_initial_properties)
649  sm_client_xsmp_set_initial_properties (xsmp);
650 
651  /* If this is the initial SaveYourself, ignore it; we've already set
652  * properties and there's no reason to actually save state too.
653  */
654  if (xsmp->expecting_initial_save_yourself)
655  {
656  xsmp->expecting_initial_save_yourself = FALSE;
657 
658  if (save_type == SmSaveLocal &&
659  interact_style == SmInteractStyleNone &&
660  !shutdown && !fast)
661  {
662  g_debug ("Sending SaveYourselfDone(True) for initial SaveYourself");
663  SmcSaveYourselfDone (xsmp->connection, True);
664  /* As explained in the comment at the end of
665  * do_save_yourself(), SAVE_YOURSELF_DONE is the correct
666  * state here, not IDLE.
667  */
668  xsmp->state = XSMP_STATE_SAVE_YOURSELF_DONE;
669  return;
670  }
671  else
672  g_warning ("First SaveYourself was not the expected one!");
673  }
674 
675  /* Even ignoring the "fast" flag completely, there are still 18
676  * different combinations of save_type, shutdown and interact_style.
677  * We interpret them as follows:
678  *
679  * Type Shutdown Interact Interpretation
680  * G F A/E/N do nothing (1)
681  * G T N do nothing (1)*
682  * G T A/E quit_requested (2)
683  * L/B F A/E/N save_state (3)
684  * L/B T N save_state (3)*
685  * L/B T A/E quit_requested, then save_state (4)
686  *
687  * 1. Do nothing, because the SM asked us to do something
688  * uninteresting (save open files, but then don't quit
689  * afterward) or rude (save open files without asking the user
690  * for confirmation).
691  *
692  * 2. Request interaction and then emit ::quit_requested. This
693  * perhaps isn't quite correct for the SmInteractStyleErrors
694  * case, but we don't care.
695  *
696  * 3. Emit ::save_state. The SmSaveBoth SaveYourselfs in these
697  * rows essentially get demoted to SmSaveLocal, because their
698  * Global halves correspond to "do nothing".
699  *
700  * 4. Request interaction, emit ::quit_requested, and then emit
701  * ::save_state after interacting. This is the SmSaveBoth
702  * equivalent of #2, but we also promote SmSaveLocal shutdown
703  * SaveYourselfs to SmSaveBoth here, because we want to give
704  * the user a chance to save open files before quitting.
705  *
706  * (* It would be nice if we could do something useful when the
707  * session manager sends a SaveYourself with shutdown True and
708  * SmInteractStyleNone. But we can't, so we just pretend it didn't
709  * even tell us it was shutting down. The docs for ::quit mention
710  * that it might not always be preceded by ::quit_requested.)
711  */
712 
713  /* As an optimization, we don't actually request interaction and
714  * emit ::quit_requested if the application isn't listening to the
715  * signal.
716  */
717  wants_quit_requested = g_signal_has_handler_pending (xsmp, g_signal_lookup ("quit_requested", EGG_TYPE_SM_CLIENT), 0, FALSE);
718 
719  xsmp->need_save_state = (save_type != SmSaveGlobal);
720  xsmp->need_quit_requested = (shutdown && wants_quit_requested &&
721  interact_style != SmInteractStyleNone);
722  xsmp->interact_errors = (interact_style == SmInteractStyleErrors);
723 
724  xsmp->shutting_down = shutdown;
725 
726  do_save_yourself (xsmp);
727 }

729 static void
730 do_save_yourself (EggSMClientXSMP *xsmp)
731 {
732  if (xsmp->state == XSMP_STATE_SHUTDOWN_CANCELLED)
733  {
734  /* The SM cancelled a previous SaveYourself, but we haven't yet
735  * had a chance to tell the application, so we can't start
736  * processing this SaveYourself yet.
737  */
738  xsmp->waiting_to_save_myself = TRUE;
739  update_pending_events (xsmp);
740  return;
741  }
742 
743  if (xsmp->need_quit_requested)
744  {
745  xsmp->state = XSMP_STATE_INTERACT_REQUEST;
746 
747  g_debug ("Sending InteractRequest(%s)",
748  xsmp->interact_errors ? "Error" : "Normal");
749  SmcInteractRequest (xsmp->connection,
750  xsmp->interact_errors ? SmDialogError : SmDialogNormal,
751  xsmp_interact,
752  xsmp);
753  return;
754  }
755 
756  if (xsmp->need_save_state)
757  {
758  save_state (xsmp);
759 
760  /* Though unlikely, the client could have been disconnected
761  * while the application was saving its state.
762  */
763  if (!xsmp->connection)
764  return;
765  }
766 
767  g_debug ("Sending SaveYourselfDone(True)");
768  SmcSaveYourselfDone (xsmp->connection, True);
769 
770  /* The client state diagram in the XSMP spec says that after a
771  * non-shutdown SaveYourself, we go directly back to "idle". But
772  * everything else in both the XSMP spec and the libSM docs
773  * disagrees.
774  */
775  xsmp->state = XSMP_STATE_SAVE_YOURSELF_DONE;
776 }

778 static void
779 save_state (EggSMClientXSMP *xsmp)
780 {
781  GKeyFile *state_file;
782  char *state_file_path, *data;
783  EggDesktopFile *desktop_file;
784  GPtrArray *restart;
785  int offset, fd;
786 
787  /* We set xsmp->state before emitting save_state, but our caller is
788  * responsible for setting it back afterward.
789  */
790  xsmp->state = XSMP_STATE_SAVE_YOURSELF;
791 
792  state_file = egg_sm_client_save_state ((EggSMClient *)xsmp);
793  if (!state_file)
794  {
795  restart = generate_command (xsmp->restart_command, xsmp->client_id, NULL);
796  set_properties (xsmp,
797  ptrarray_prop (SmRestartCommand, restart),
798  NULL);
799  g_ptr_array_free (restart, TRUE);
800  delete_properties (xsmp, SmDiscardCommand, NULL);
801  return;
802  }
803 
804  desktop_file = egg_get_desktop_file ();
805  if (desktop_file)
806  {
807  GKeyFile *merged_file;
808  char *desktop_file_path;
809 
810  merged_file = g_key_file_new ();
811  desktop_file_path =
812  g_filename_from_uri (egg_desktop_file_get_source (desktop_file),
813  NULL, NULL);
814  if (desktop_file_path &&
815  g_key_file_load_from_file (merged_file, desktop_file_path,
816  G_KEY_FILE_KEEP_COMMENTS |
817  G_KEY_FILE_KEEP_TRANSLATIONS, NULL))
818  {
819  guint g, k, i;
820  char **groups, **keys, *value, *exec;
821 
822  groups = g_key_file_get_groups (state_file, NULL);
823  for (g = 0; groups[g]; g++)
824  {
825  keys = g_key_file_get_keys (state_file, groups[g], NULL, NULL);
826  for (k = 0; keys[k]; k++)
827  {
828  value = g_key_file_get_value (state_file, groups[g],
829  keys[k], NULL);
830  if (value)
831  {
832  g_key_file_set_value (merged_file, groups[g],
833  keys[k], value);
834  g_free (value);
835  }
836  }
837  g_strfreev (keys);
838  }
839  g_strfreev (groups);
840 
841  g_key_file_free (state_file);
842  state_file = merged_file;
843 
844  /* Update Exec key using "--sm-client-state-file %k" */
845  restart = generate_command (xsmp->restart_command,
846  NULL, "%k");
847  for (i = 0; i < restart->len; i++)
848  restart->pdata[i] = g_shell_quote (restart->pdata[i]);
849  g_ptr_array_add (restart, NULL);
850  exec = g_strjoinv (" ", (char **)restart->pdata);
851  g_strfreev ((char **)restart->pdata);
852  g_ptr_array_free (restart, FALSE);
853 
854  g_key_file_set_string (state_file, EGG_DESKTOP_FILE_GROUP,
855  EGG_DESKTOP_FILE_KEY_EXEC,
856  exec);
857  g_free (exec);
858  }
859  else
860  desktop_file = NULL;
861 
862  g_free (desktop_file_path);
863  }
864 
865  /* Now write state_file to disk. (We can't use mktemp(), because
866  * that requires the filename to end with "XXXXXX", and we want
867  * it to end with ".desktop".)
868  */
869 
870  data = g_key_file_to_data (state_file, NULL, NULL);
871  g_key_file_free (state_file);
872 
873  offset = 0;
874  while (1)
875  {
876  state_file_path = g_strdup_printf ("%s%csession-state%c%s-%ld.%s",
877  g_get_user_config_dir (),
878  G_DIR_SEPARATOR, G_DIR_SEPARATOR,
879  g_get_prgname (),
880  (long)time (NULL) + offset,
881  desktop_file ? "desktop" : "state");
882 
883  fd = open (state_file_path, O_WRONLY | O_CREAT | O_EXCL, 0644);
884  if (fd == -1)
885  {
886  if (errno == EEXIST)
887  {
888  offset++;
889  g_free (state_file_path);
890  continue;
891  }
892  else if (errno == ENOTDIR || errno == ENOENT)
893  {
894  char *sep = strrchr (state_file_path, G_DIR_SEPARATOR);
895 
896  *sep = '\0';
897  if (g_mkdir_with_parents (state_file_path, 0755) != 0)
898  {
899  g_warning ("Could not create directory '%s'",
900  state_file_path);
901  g_free (state_file_path);
902  state_file_path = NULL;
903  break;
904  }
905 
906  continue;
907  }
908 
909  g_warning ("Could not create file '%s': %s",
910  state_file_path, g_strerror (errno));
911  g_free (state_file_path);
912  state_file_path = NULL;
913  break;
914  }
915 
916  close (fd);
917  g_file_set_contents (state_file_path, data, -1, NULL);
918  break;
919  }
920  g_free (data);
921 
922  restart = generate_command (xsmp->restart_command, xsmp->client_id,
923  state_file_path);
924  set_properties (xsmp,
925  ptrarray_prop (SmRestartCommand, restart),
926  NULL);
927  g_ptr_array_free (restart, TRUE);
928 
929  if (state_file_path)
930  {
931  set_properties (xsmp,
932  array_prop (SmDiscardCommand,
933  "/bin/rm", "-rf", state_file_path,
934  NULL),
935  NULL);
936  g_free (state_file_path);
937  }
938 }

989 static void
990 xsmp_shutdown_cancelled (SmcConn smc_conn,
991  SmPointer client_data)
992 {
993  EggSMClientXSMP *xsmp = client_data;
994  EggSMClient *client = client_data;
995 
996  g_debug ("Received ShutdownCancelled message in state %s",
997  EGG_SM_CLIENT_XSMP_STATE (xsmp));
998 
999  xsmp->shutting_down = FALSE;
1000 
1001  if (xsmp->state == XSMP_STATE_SAVE_YOURSELF_DONE)
1002  {
1003  /* We've finished interacting and now the SM has agreed to
1004  * cancel the shutdown.
1005  */
1006  xsmp->state = XSMP_STATE_IDLE;
1007  egg_sm_client_quit_cancelled (client);
1008  }
1009  else if (xsmp->state == XSMP_STATE_SHUTDOWN_CANCELLED)
1010  {
1011  /* Hm... ok, so we got a shutdown SaveYourself, which got
1012  * cancelled, but the application was still interacting, so we
1013  * didn't tell it yet, and then *another* SaveYourself arrived,
1014  * which we must still be waiting to tell the app about, except
1015  * that now that SaveYourself has been cancelled too! Dizzy yet?
1016  */
1017  xsmp->waiting_to_save_myself = FALSE;
1018  update_pending_events (xsmp);
1019  }
1020  else
1021  {
1022  g_debug ("Sending SaveYourselfDone(False)");
1023  SmcSaveYourselfDone (xsmp->connection, False);
1024 
1025  if (xsmp->state == XSMP_STATE_INTERACT)
1026  {
1027  /* The application is currently interacting, so we can't
1028  * tell it about the cancellation yet; we will wait until
1029  * after it calls egg_sm_client_will_quit().
1030  */
1031  xsmp->state = XSMP_STATE_SHUTDOWN_CANCELLED;
1032  }
1033  else
1034  {
1035  /* The shutdown was cancelled before the application got a
1036  * chance to interact.
1037  */
1038  xsmp->state = XSMP_STATE_IDLE;
1039  }
1040  }
1041 }
