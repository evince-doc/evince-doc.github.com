80 static void
81 handle_message (const ddjvu_message_t *msg, GError **error)
82 {
83  switch (msg->m_any.tag) {
84  case DDJVU_ERROR: {
85  gchar *error_str;
86 
87  if (msg->m_error.filename) {
88  error_str = g_strdup_printf ("DjvuLibre error: %s:%d",
89  msg->m_error.filename,
90  msg->m_error.lineno);
91  } else {
92  error_str = g_strdup_printf ("DjvuLibre error: %s",
93  msg->m_error.message);
94  }
95 
96  if (error) {
97  g_set_error_literal (error, EV_DJVU_ERROR, 0, error_str);
98  } else {
99  g_warning ("%s", error_str);
100  }
101 
102  g_free (error_str);
103  return;
104  }
105  break;
106  default:
107  break;
108  }

111 void
112 djvu_handle_events (DjvuDocument *djvu_document, int wait, GError **error)
113 {
114  ddjvu_context_t *ctx = djvu_document->d_context;
115  const ddjvu_message_t *msg;
116 
117  if (!ctx)
118  return;
119 
120  if (wait)
121  ddjvu_message_wait (ctx);
122 
123  while ((msg = ddjvu_message_peek (ctx))) {
124  handle_message (msg, error);
125  ddjvu_message_pop (ctx);
126  if (error && *error)
127  return;
128  }
129 }

131 static void
132 djvu_wait_for_message (DjvuDocument *djvu_document, ddjvu_message_tag_t message, GError **error)
133 {
134  ddjvu_context_t *ctx = djvu_document->d_context;
135  const ddjvu_message_t *msg;
136 
137  ddjvu_message_wait (ctx);
138  while ((msg = ddjvu_message_peek (ctx)) && (msg->m_any.tag != message)) {
139  handle_message (msg, error);
140  ddjvu_message_pop (ctx);

142  return;

144  if (msg && msg->m_any.tag == message)

148 static gboolean
149 djvu_document_load (EvDocument *document,
150  const char *uri,
151  GError **error)
152 {
153  DjvuDocument *djvu_document = DJVU_DOCUMENT (document);
154  ddjvu_document_t *doc;
155  gchar *filename;
156  gboolean missing_files = FALSE;
157  gboolean check_for_missing_files = FALSE;
158  GError *djvu_error = NULL;
159  gint n_files;
160  gint i;
161  gchar *base;
162 
163  /* FIXME: We could actually load uris */
164  filename = g_filename_from_uri (uri, NULL, error);
165  if (!filename)
166  return FALSE;
167 
168  doc = ddjvu_document_create_by_filename (djvu_document->d_context, filename, TRUE);
169 
170  if (!doc) {
171  g_free (filename);
172  g_set_error_literal (error,
173  EV_DOCUMENT_ERROR,
174  EV_DOCUMENT_ERROR_INVALID,
175  _("DjVu document has incorrect format"));
176  return FALSE;
177  }
178 
179  if (djvu_document->d_document)
180  ddjvu_document_release (djvu_document->d_document);
181 
182  djvu_document->d_document = doc;
183 
184  djvu_wait_for_message (djvu_document, DDJVU_DOCINFO, &djvu_error);
185  if (djvu_error) {
186  g_set_error_literal (error,
187  EV_DOCUMENT_ERROR,
188  EV_DOCUMENT_ERROR_INVALID,
189  djvu_error->message);
190  g_error_free (djvu_error);
191  g_free (filename);
192  ddjvu_document_release (djvu_document->d_document);
193  djvu_document->d_document = NULL;
194 
195  return FALSE;
196  }
197 
198  if (ddjvu_document_decoding_error (djvu_document->d_document))
199  djvu_handle_events (djvu_document, TRUE, &djvu_error);
200 
201  if (djvu_error) {
202  g_set_error_literal (error,
203  EV_DOCUMENT_ERROR,
204  EV_DOCUMENT_ERROR_INVALID,
205  djvu_error->message);
206  g_error_free (djvu_error);
207  g_free (filename);
208  ddjvu_document_release (djvu_document->d_document);
209  djvu_document->d_document = NULL;
210 
211  return FALSE;
212  }
213 
214  g_free (djvu_document->uri);
215  djvu_document->uri = g_strdup (uri);
216 
217  djvu_document->n_pages = ddjvu_document_get_pagenum (djvu_document->d_document);
218 
219  if (djvu_document->n_pages > 0) {
220  djvu_document->fileinfo_pages = g_new0 (ddjvu_fileinfo_t, djvu_document->n_pages);
221  djvu_document->file_ids = g_hash_table_new (g_str_hash, g_str_equal);
222  }
223  if (ddjvu_document_get_type (djvu_document->d_document) == DDJVU_DOCTYPE_INDIRECT)
224  check_for_missing_files = TRUE;
225 
226  base = g_path_get_dirname (filename);
227 
228  n_files = ddjvu_document_get_filenum (djvu_document->d_document);
229  for (i = 0; i < n_files; i++) {
230  ddjvu_fileinfo_t fileinfo;
231  gchar *file;
232 
233  ddjvu_document_get_fileinfo (djvu_document->d_document,
234  i, &fileinfo);
235 
236  if (fileinfo.type != 'P')
237  continue;
238 
239  if (fileinfo.pageno >= 0 && fileinfo.pageno < djvu_document->n_pages) {
240  djvu_document->fileinfo_pages[fileinfo.pageno] = fileinfo;
241  }
242 
243  g_hash_table_insert (djvu_document->file_ids,
244  (gpointer) djvu_document->fileinfo_pages[fileinfo.pageno].id,
245  GINT_TO_POINTER(fileinfo.pageno));
246 
247  if (check_for_missing_files && !missing_files) {
248  file = g_build_filename (base, fileinfo.id, NULL);
249  if (!g_file_test (file, G_FILE_TEST_EXISTS)) {
250  missing_files = TRUE;
251  }
252  g_free (file);
253  }
254  }
255  g_free (base);
256  g_free (filename);
257 
258  if (missing_files) {
259  g_set_error_literal (error,
260  G_FILE_ERROR,
261  G_FILE_ERROR_EXIST,
262  _("The document is composed of several files. "
263  "One or more of these files cannot be accessed."));
264 
265  return FALSE;
266  }
267 
268  return TRUE;
269 }

292 static void
293 document_get_page_size (DjvuDocument *djvu_document,
294  gint page,
295  double *width,
296  double *height,
297  double *dpi)
298 {
299  ddjvu_pageinfo_t info;
300  ddjvu_status_t r;
301 
302  while ((r = ddjvu_document_get_pageinfo(djvu_document->d_document, page, &info)) < DDJVU_JOB_OK)
303  djvu_handle_events(djvu_document, TRUE, NULL);
304 
305  if (r >= DDJVU_JOB_FAILED)
306  djvu_handle_events(djvu_document, TRUE, NULL);
307 
308  if (width)
309  *width = info.width * 72.0 / info.dpi;
310  if (height)
311  *height = info.height * 72.0 / info.dpi;
312  if (dpi)
313  *dpi = info.dpi;
314 }

330 static cairo_surface_t *
331 djvu_document_render (EvDocument *document,
332  EvRenderContext *rc)
333 {
334  DjvuDocument *djvu_document = DJVU_DOCUMENT (document);
335  cairo_surface_t *surface;
336  gchar *pixels;
337  gint rowstride;
338  ddjvu_rect_t rrect;
339  ddjvu_rect_t prect;
340  ddjvu_page_t *d_page;
341  ddjvu_page_rotation_t rotation;
342  gint buffer_modified;
343  double page_width, page_height;
344  gint transformed_width, transformed_height;
345 
346  d_page = ddjvu_page_create_by_pageno (djvu_document->d_document, rc->page->index);
347 
348  while (!ddjvu_page_decoding_done (d_page))
349  djvu_handle_events(djvu_document, TRUE, NULL);
350 
351  document_get_page_size (djvu_document, rc->page->index, &page_width, &page_height, NULL);
352  rotation = ddjvu_page_get_initial_rotation (d_page);
353 
354  ev_render_context_compute_transformed_size (rc, page_width, page_height,
355  &transformed_width, &transformed_height);
356 
357  /*
358  * Evince rotates documents in clockwise direction
359  * and djvulibre rotates documents in counter-clockwise direction.
360  */
361  switch (rc->rotation) {
362  case 90:
363  rotation += DDJVU_ROTATE_270;
364 
365  break;
366  case 180:
367  rotation += DDJVU_ROTATE_180;
368 
369  break;
370  case 270:
371  rotation += DDJVU_ROTATE_90;
372 
373  break;
374  default:
375  rotation += DDJVU_ROTATE_0;
376  }
377  rotation = rotation % 4;
378 
379  surface = cairo_image_surface_create (CAIRO_FORMAT_RGB24,
380  transformed_width, transformed_height);
381 
382  rowstride = cairo_image_surface_get_stride (surface);
383  pixels = (gchar *)cairo_image_surface_get_data (surface);
384 
385  prect.x = 0;
386  prect.y = 0;
387  prect.w = transformed_width;
388  prect.h = transformed_height;
389  rrect = prect;
390 
391  ddjvu_page_set_rotation (d_page, rotation);
392 
393  buffer_modified = ddjvu_page_render (d_page, DDJVU_RENDER_COLOR,
394  &prect,
395  &rrect,
396  djvu_document->d_format,
397  rowstride,
398  pixels);
399 
400  if (!buffer_modified) {
401  cairo_t *cr = cairo_create (surface);
402 
403  cairo_set_source_rgb (cr, 1.0, 1.0, 1.0);
404  cairo_paint (cr);
405  cairo_destroy (cr);
406  } else {
407  cairo_surface_mark_dirty (surface);
408  }
409 
410  return surface;
411 }

645  region = cairo_region_create ();
646  for (l = rects; l && l->data; l = g_list_next (l)) {
647  cairo_rectangle_int_t rect;
648  EvRectangle *r = (EvRectangle *)l->data;
649  gdouble tmp;
650 
651  tmp = r->y1;
652  r->x1 *= 72 / dpi;
653  r->x2 *= 72 / dpi;
654  r->y1 = height - r->y2 * 72 / dpi;
655  r->y2 = height - tmp * 72 / dpi;
656 
657  rect.x = (gint) ((r->x1 * scale_x) + 0.5);
658  rect.y = (gint) ((r->y1 * scale_y) + 0.5);
659  rect.width = (gint) ((r->x2 * scale_x) + 0.5) - rect.x;
660  rect.height = (gint) ((r->y2 * scale_y) + 0.5) - rect.y;
661  cairo_region_union_rectangle (region, &rect);
662  ev_rectangle_free (r);
663  }
664  g_list_free (l);
665 
666  return region;
667 }

848 static GList *
849 djvu_document_find_find_text (EvDocumentFind *document,
850  EvPage *page,
851  const char *text,
852  gboolean case_sensitive)
853 {
854  DjvuDocument *djvu_document = DJVU_DOCUMENT (document);
855  miniexp_t page_text;
856  gdouble width, height, dpi;
857  GList *matches = NULL, *l;
858 
859  g_return_val_if_fail (text != NULL, NULL);
860 
861  while ((page_text = ddjvu_document_get_pagetext (djvu_document->d_document,
862  page->index,
863  "char")) == miniexp_dummy)
864  djvu_handle_events (djvu_document, TRUE, NULL);
865 
866  if (page_text != miniexp_nil) {
867  DjvuTextPage *tpage = djvu_text_page_new (page_text);
868 
869  djvu_text_page_index_text (tpage, case_sensitive);
870  if (tpage->links->len > 0) {
871  djvu_text_page_search (tpage, text);
872  matches = tpage->results;
873  }
874  djvu_text_page_free (tpage);
875  ddjvu_miniexp_release (djvu_document->d_document, page_text);
876  }
877  if (!matches)
878  return NULL;
879 
880  document_get_page_size (djvu_document, page->index, &width, &height, &dpi);
881  for (l = matches; l && l->data; l = g_list_next (l)) {
882  EvRectangle *r = (EvRectangle *)l->data;
883  gdouble tmp = r->y1;
884 
885  r->x1 *= 72.0 / dpi;
886  r->x2 *= 72.0 / dpi;
887 
888  r->y1 = height - r->y2 * 72.0 / dpi;
889  r->y2 = height - tmp * 72.0 / dpi;
890  }
891 
892 
893  return matches;
894 }
