80 static EvDocument *
81 ev_document_factory_new_document_for_mime_type (const gchar *mime_type,
82  GError **error)
83 {
84  EvDocument *document;
85  EvBackendInfo *info;
86  GTypeModule *module = NULL;
87 
88  g_return_val_if_fail (mime_type != NULL, NULL);
89 
90  info = get_backend_info_for_mime_type (mime_type);
91  if (info == NULL) {
92  char *content_type, *mime_desc = NULL;
93 
94  content_type = g_content_type_from_mime_type (mime_type);
95  if (content_type)
96  mime_desc = g_content_type_get_description (content_type);
97 
98  g_set_error (error,
99  EV_DOCUMENT_ERROR,
100  EV_DOCUMENT_ERROR_INVALID,
101  _("File type %s (%s) is not supported"),
102  mime_desc ? mime_desc : "(unknown)", mime_type);
103  g_free (mime_desc);
104  g_free (content_type);
105 
106  return NULL;
107  }
108 
109  if (ev_module_hash != NULL) {
110  module = g_hash_table_lookup (ev_module_hash, info->module_name);
111  }
112  if (module == NULL) {
113  gchar *path;
114 
115  path = g_module_build_path (ev_backends_dir, info->module_name);
116  module = G_TYPE_MODULE (_ev_module_new (path, info->resident));
117  g_free (path);
118 
119  if (ev_module_hash == NULL) {
120  ev_module_hash = g_hash_table_new_full (g_str_hash, g_str_equal,
121  g_free,
122  NULL /* leaked on purpose */);
123  }
124  g_hash_table_insert (ev_module_hash, g_strdup (info->module_name), module);
125  }
126 
127  if (!g_type_module_use (module)) {
128  const char *err;
129 
130  err = g_module_error ();
131  g_set_error (error, EV_DOCUMENT_ERROR, EV_DOCUMENT_ERROR_INVALID,
132  "Failed to load backend for '%s': %s",
133  mime_type, err ? err : "unknown error");
134  return NULL;
135  }
136 
137  document = EV_DOCUMENT (_ev_module_new_object (EV_MODULE (module)));
138  g_type_module_unuse (module);
139 
140  g_object_set_data_full (G_OBJECT (document), BACKEND_DATA_KEY,
141  _ev_backend_info_ref (info),
142  (GDestroyNotify) _ev_backend_info_unref);
143 
144  return document;
145 }

147 static EvCompressionType
148 get_compression_from_mime_type (const gchar *mime_type)
149 {
150  gchar type[3];
151  gchar *p;
152 
153  if (!(p = g_strrstr (mime_type, "/")))
154  return EV_COMPRESSION_NONE;
155 
156  if (sscanf (++p, "x-%2s%*s", type) == 1) {
157  if (g_ascii_strcasecmp (type, "gz") == 0)
158  return EV_COMPRESSION_GZIP;
159  else if (g_ascii_strcasecmp (type, "bz") == 0)
160  return EV_COMPRESSION_BZIP2;
161  else if (g_ascii_strcasecmp (type, "xz") == 0)
162  return EV_COMPRESSION_LZMA;
163  }
164 
165  return EV_COMPRESSION_NONE;
166 }

169 /*
170  * new_document_for_uri:
171  * @uri: the document URI
172  * @fast: whether to use fast MIME type detection
173  * @compression: a location to store the document's compression type
174  * @error: a #GError location to store an error, or %NULL
175  *
176  * Creates a #EvDocument instance for the document at @uri, using either
177  * fast or slow MIME type detection. If a document could be created,
178  * @compression is filled in with the document's compression type.
179  * On error, %NULL is returned and @error filled in.
180  *
181  * Returns: a new #EvDocument instance, or %NULL on error with @error filled in
182  */
183 static EvDocument *
184 new_document_for_uri (const char *uri,
185  gboolean fast,
186  EvCompressionType *compression,
187  GError **error)
188 {
189  EvDocument *document = NULL;
190  gchar *mime_type = NULL;
191 
192  *compression = EV_COMPRESSION_NONE;
193 
194  mime_type = ev_file_get_mime_type (uri, fast, error);
195  if (mime_type == NULL)
196  return NULL;
197 
198  document = ev_document_factory_new_document_for_mime_type (mime_type, error);
199  if (document == NULL)
200  return NULL;
201 
202  *compression = get_compression_from_mime_type (mime_type);
203 
204  g_free (mime_type);
205 
206  return document;
207 }

285 EvDocument *
286 ev_document_factory_get_document (const char *uri, GError **error)
287 {
288  EvDocument *document;
289  int result;
290  EvCompressionType compression;
291  gchar *uri_unc = NULL;
292  GError *err = NULL;
293 
294  g_return_val_if_fail (uri != NULL, NULL);
295 
296  document = new_document_for_uri (uri, TRUE, &compression, &err);
297  g_assert (document != NULL || err != NULL);
298 
299  if (document != NULL) {
300  uri_unc = ev_file_uncompress (uri, compression, &err);
301  if (uri_unc) {
302  g_object_set_data_full (G_OBJECT (document),
303  "uri-uncompressed",
304  uri_unc,
305  (GDestroyNotify) free_uncompressed_uri);
306  } else if (err != NULL) {
307  /* Error uncompressing file */
308  g_object_unref (document);
309  g_propagate_error (error, err);
310  return NULL;
311  }
312 
313  result = ev_document_load (document, uri_unc ? uri_unc : uri, &err);
314 
315  if (result == FALSE || err) {
316  if (err &&
317  g_error_matches (err, EV_DOCUMENT_ERROR, EV_DOCUMENT_ERROR_ENCRYPTED)) {
318  g_propagate_error (error, err);
319  return document;
320  }
321  /* else fall through to slow mime code section below */
322  } else {
323  return document;
324  }
325 
326  g_object_unref (document);
327  document = NULL;
328  }
329 
330  /* Try again with slow mime detection */
331  g_clear_error (&err);
332  uri_unc = NULL;
333 
334  document = new_document_for_uri (uri, FALSE, &compression, &err);
335  if (document == NULL) {
336  g_assert (err != NULL);
337  g_propagate_error (error, err);
338  return NULL;
339  }
340 
341  uri_unc = ev_file_uncompress (uri, compression, &err);
342  if (uri_unc) {
343  g_object_set_data_full (G_OBJECT (document),
344  "uri-uncompressed",
345  uri_unc,
346  (GDestroyNotify) free_uncompressed_uri);
347  } else if (err != NULL) {
348  /* Error uncompressing file */
349  g_propagate_error (error, err);
350 
351  g_object_unref (document);
352  return NULL;
353  }
354 
355  result = ev_document_load (document, uri_unc ? uri_unc : uri, &err);
356  if (result == FALSE) {
357  if (err == NULL) {
358  /* FIXME: this really should not happen; the backend should
359  * always return a meaningful error.
360  */
361  g_set_error_literal (&err,
362  EV_DOCUMENT_ERROR,
363  EV_DOCUMENT_ERROR_INVALID,
364  _("Unknown MIME Type"));
365  } else if (g_error_matches (err, EV_DOCUMENT_ERROR, EV_DOCUMENT_ERROR_ENCRYPTED)) {
366  g_propagate_error (error, err);
367  return document;
368  }
369 
370  g_object_unref (document);
371  document = NULL;
372 
373  g_propagate_error (error, err);
374  }
375 
376  return document;
377 }

397 ev_document_factory_get_document_for_gfile (GFile *file,
398  EvDocumentLoadFlags flags,
399  GCancellable *cancellable,
400  GError **error)
401 {
402  EvDocument *document;
403  GFileInfo *file_info;
404  const char *content_type;
405  char *mime_type = NULL;
406 
407  g_return_val_if_fail (G_IS_FILE (file), NULL);
408  g_return_val_if_fail (error == NULL || *error == NULL, NULL);
409 
410 
411  file_info = g_file_query_info (file,
412  G_FILE_ATTRIBUTE_STANDARD_CONTENT_TYPE,

414  cancellable,
415  error);

417  return NULL;
418 
419  content_type = g_file_info_get_content_type (file_info);
420  if (content_type == NULL) {
421  g_set_error_literal (error, G_IO_ERROR, G_IO_ERROR_FAILED,
422  "Failed to query file mime type");
423  return NULL;
424  }
425 
426  mime_type = g_content_type_get_mime_type (content_type);
427  g_object_unref (file_info);
428 
429  document = ev_document_factory_new_document_for_mime_type (mime_type, error);
430  g_free (mime_type);
431  if (document == NULL)
432  return NULL;
433 
434  if (!ev_document_load_gfile (document, file, flags, cancellable, error)) {
435  g_object_unref (document);
436  return NULL;

465 ev_document_factory_get_document_for_stream (GInputStream *stream,
466  const char *mime_type,

468  GCancellable *cancellable,
469  GError **error)
470 {
471  EvDocument *document;

473 
474  g_return_val_if_fail (G_IS_INPUT_STREAM (stream), NULL);
475  g_return_val_if_fail (error == NULL || *error == NULL, NULL);
476 
477  if (mime_type == NULL && G_IS_FILE_INPUT_STREAM (stream)) {
478  GFileInfo *file_info;
479  const char *content_type;
480 
481  file_info = g_file_input_stream_query_info (G_FILE_INPUT_STREAM (stream),
482  G_FILE_ATTRIBUTE_STANDARD_CONTENT_TYPE,
483  cancellable,
484  error);
485  if (file_info != NULL) {
486  content_type = g_file_info_get_content_type (file_info);
487  if (content_type)
488  mime_type = mime = g_content_type_get_mime_type (content_type);
489  g_object_unref (file_info);
490  }
491  }
492 
493  if (mime_type == NULL) {
494  g_set_error_literal (error, G_IO_ERROR, G_IO_ERROR_NOT_SUPPORTED,
495  "Cannot query mime type from stream");
496  return NULL;
497  }
498 
499  document = ev_document_factory_new_document_for_mime_type (mime_type, error);
500  g_free (mime);
501 
502  if (document == NULL)
503  return NULL;
504 
505  if (!ev_document_load_stream (document, stream, flags, cancellable, error)) {
506  g_object_unref (document);
507  return NULL;
508  }
509 
510  return document;
511 }

541 void
542 ev_document_factory_add_filters (GtkWidget *chooser, EvDocument *document)
543 {
544  GtkFileFilter *filter;
545  GtkFileFilter *default_filter;
546  GtkFileFilter *document_filter;
547 
548  g_return_if_fail (GTK_IS_FILE_CHOOSER (chooser));
549  g_return_if_fail (document == NULL || EV_IS_DOCUMENT (document));
550 
551  default_filter = document_filter = filter = gtk_file_filter_new ();
552  gtk_file_filter_set_name (filter, _("All Documents"));
553  g_list_foreach (ev_backends_list, (GFunc)file_filter_add_mime_types, filter);
554  gtk_file_chooser_add_filter (GTK_FILE_CHOOSER (chooser), filter);
555 
556  if (document) {
557  EvBackendInfo *info;
558 
559  info = get_backend_info_for_document (document);
560  g_assert (info != NULL);
561  default_filter = filter = gtk_file_filter_new ();
562  gtk_file_filter_set_name (filter, info->type_desc);
563  file_filter_add_mime_types (info, filter);
564  gtk_file_chooser_add_filter (GTK_FILE_CHOOSER (chooser), filter);
565  } else {
566  GList *l;
567 
568  for (l = ev_backends_list; l; l = l->next) {
569  EvBackendInfo *info = (EvBackendInfo *) l->data;
570 
571  default_filter = filter = gtk_file_filter_new ();
572  gtk_file_filter_set_name (filter, info->type_desc);
573  file_filter_add_mime_types (info, filter);
574  gtk_file_chooser_add_filter (GTK_FILE_CHOOSER (chooser), filter);
575  }
576  }
577 
578  filter = gtk_file_filter_new ();
579  gtk_file_filter_set_name (filter, _("All Files"));
580  gtk_file_filter_add_pattern (filter, "*");
581  gtk_file_chooser_add_filter (GTK_FILE_CHOOSER (chooser), filter);
582 
583  gtk_file_chooser_set_filter (GTK_FILE_CHOOSER (chooser),
584  document == NULL ? document_filter : default_filter);
585 }
