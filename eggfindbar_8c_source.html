584 void
585 egg_find_bar_set_search_string (EggFindBar *find_bar,
586  const char *search_string)
587 {
588  EggFindBarPrivate *priv;
589 
590  g_return_if_fail (EGG_IS_FIND_BAR (find_bar));
591 
592  priv = (EggFindBarPrivate *)find_bar->priv;
593 
594  g_object_freeze_notify (G_OBJECT (find_bar));
595 
596  if (priv->search_string != search_string)
597  {
598  char *old;
599 
600  old = priv->search_string;
601 
602  if (search_string && *search_string == '\0')
603  search_string = NULL;
604 
605  /* Only update if the string has changed; setting the entry
606  * will emit changed on the entry which will re-enter
607  * this function, but we'll handle that fine with this
608  * short-circuit.
609  */
610  if ((old && search_string == NULL) ||
611  (old == NULL && search_string) ||
612  (old && search_string &&
613  strcmp (old, search_string) != 0))
614  {
615  gboolean not_empty;
616 
617  priv->search_string = g_strdup (search_string);
618  g_free (old);
619 
620  gtk_entry_set_text (GTK_ENTRY (priv->find_entry),
621  priv->search_string ?
622  priv->search_string :
623  "");
624 
625  not_empty = (search_string == NULL) ? FALSE : TRUE;
626 
627  gtk_widget_set_sensitive (GTK_WIDGET (find_bar->priv->next_button), not_empty);
628  gtk_widget_set_sensitive (GTK_WIDGET (find_bar->priv->previous_button), not_empty);
629 
630  g_object_notify (G_OBJECT (find_bar),
631  "search_string");
632  }
633  }
634 
635  g_object_thaw_notify (G_OBJECT (find_bar));
636 }
