108  : m_instance(instance)

140 static void sizingModeChanged(EvDocumentModel *model, GParamSpec *, EvBrowserPluginToolbar *toolbar)
141 {
142  {
143  SignalBlocker fitPageBlocker(toolbar->priv->zoomFitPageRadioButton, zoomFitPageToggled, toolbar);
144  gtk_check_menu_item_set_active(GTK_CHECK_MENU_ITEM(toolbar->priv->zoomFitPageRadioButton),
145  toolbar->priv->plugin->sizingMode() == EV_SIZING_FIT_PAGE);

147

151  toolbar->priv->plugin->sizingMode() == EV_SIZING_FIT_WIDTH);

153

161 static void evBrowserPluginToolbarSetProperty(GObject *object, guint propID, const GValue *value, GParamSpec *paramSpec)
162 {
163  EvBrowserPluginToolbar *toolbar = EV_BROWSER_PLUGIN_TOOLBAR(object);

263 static void evBrowserPluginToolbarConstructed(GObject *object)
264 {
265  G_OBJECT_CLASS(ev_browser_plugin_toolbar_parent_class)->constructed(object);
266 
267  EvBrowserPluginToolbar *toolbar = EV_BROWSER_PLUGIN_TOOLBAR(object);
268  bool rtl = gtk_widget_get_direction(GTK_WIDGET(toolbar)) == GTK_TEXT_DIR_RTL;
269 
270  GtkWidget *hbox = createButtonGroup(toolbar);
271 
272  // Navigation buttons
273  GtkWidget *button = createButton(toolbar, "go-up-symbolic", _("Go to the previous page"), G_CALLBACK(goToPreviousPage));
274  gtk_container_add(GTK_CONTAINER(hbox), button);
275  gtk_widget_show(button);
276 
277  button = createButton(toolbar, "go-down-symbolic", _("Go to the next page"), G_CALLBACK(goToNextPage));
278  gtk_container_add(GTK_CONTAINER(hbox), button);
279  gtk_widget_show(button);
280 
281  GtkWidget *toolItem = GTK_WIDGET(gtk_tool_item_new());
282  if (rtl)
283  gtk_widget_set_margin_left(toolItem, 12);
284  else
285  gtk_widget_set_margin_right(toolItem, 12);
286  gtk_container_add(GTK_CONTAINER(toolItem), hbox);
287  gtk_widget_show(hbox);
288 
289  gtk_container_add(GTK_CONTAINER(toolbar), toolItem);
290  gtk_widget_show(toolItem);
291 
292  // Page Selector
293  toolItem = GTK_WIDGET(g_object_new(EV_TYPE_PAGE_ACTION_WIDGET, nullptr));
294  ev_page_action_widget_set_model(EV_PAGE_ACTION_WIDGET(toolItem), toolbar->priv->plugin->model());
295  g_signal_connect_swapped(toolItem, "activate-link", G_CALLBACK(activateLink), toolbar);
296  if (rtl)
297  gtk_widget_set_margin_left(toolItem, 12);
298  else
299  gtk_widget_set_margin_right(toolItem, 12);
300  gtk_container_add(GTK_CONTAINER(toolbar), toolItem);
301  gtk_widget_show(toolItem);
302 
303  // Separator
304  toolItem = GTK_WIDGET(gtk_tool_item_new());
305  gtk_tool_item_set_expand(GTK_TOOL_ITEM(toolItem), TRUE);
306  gtk_container_add(GTK_CONTAINER(toolbar), toolItem);
307  gtk_widget_show(toolItem);
308 
309  // View mode
310  hbox = createButtonGroup(toolbar);
311 
312  button = createToggleButton(toolbar, "view-continuous-symbolic", _("Show the entire document"),
313  toolbar->priv->plugin->isContinuous(), G_CALLBACK(toggleContinuous));
314  toolbar->priv->continuousToggleButton = button;
315  g_signal_connect(toolbar->priv->plugin->model(), "notify::continuous",
316  G_CALLBACK(continuousChanged), toolbar);
317  gtk_container_add(GTK_CONTAINER(hbox), button);
318  gtk_widget_show(button);
319 
320  button = createToggleButton(toolbar, "view-dual-symbolic", _("Show two pages at once"),
321  toolbar->priv->plugin->isDual(), G_CALLBACK(toggleDual));
322  toolbar->priv->dualToggleButton = button;
323  g_signal_connect(toolbar->priv->plugin->model(), "notify::dual-page",
324  G_CALLBACK(dualPageChanged), toolbar);
325  gtk_container_add(GTK_CONTAINER(hbox), button);
326  gtk_widget_show(button);
327 
328  toolItem = GTK_WIDGET(gtk_tool_item_new());
329  if (rtl)
330  gtk_widget_set_margin_left(toolItem, 12);
331  else
332  gtk_widget_set_margin_right(toolItem, 12);
333  gtk_container_add(GTK_CONTAINER(toolItem), hbox);
334  gtk_widget_show(hbox);
335 
336  gtk_container_add(GTK_CONTAINER(toolbar), toolItem);
337  gtk_widget_show(toolItem);
338 
339  // Zoom
340  hbox = createButtonGroup(toolbar);
341 
342  button = createButton(toolbar, "zoom-in-symbolic", _("Enlarge the document"), G_CALLBACK(zoomIn));
343  gtk_container_add(GTK_CONTAINER(hbox), button);
344  gtk_widget_show(button);
345 
346  button = createButton(toolbar, "zoom-out-symbolic", _("Shrink the document"), G_CALLBACK(zoomOut));
347  gtk_container_add(GTK_CONTAINER(hbox), button);
348  gtk_widget_show(button);
349 
350  button = createMenuButton(toolbar, "go-down-symbolic", createSizingModeMenu(toolbar), GTK_ALIGN_END);
351  gtk_container_add(GTK_CONTAINER(hbox), button);
352  gtk_widget_show(button);
353 
354  toolItem = GTK_WIDGET(gtk_tool_item_new());
355  if (rtl)
356  gtk_widget_set_margin_left(toolItem, 12);
357  else
358  gtk_widget_set_margin_right(toolItem, 12);
359  gtk_container_add(GTK_CONTAINER(toolItem), hbox);
360  gtk_widget_show(hbox);
361 
362  gtk_container_add(GTK_CONTAINER(toolbar), toolItem);
363  gtk_widget_show(toolItem);
364 
365  // Actions
366  // Only add download button if browser is Epiphany for now.
367  if (toolbar->priv->plugin->canDownload()) {
368  button = createButton(toolbar, "folder-download-symbolic", _("Download document"), G_CALLBACK(downloadDocument));
369  toolItem = GTK_WIDGET(gtk_tool_item_new());
370  gtk_container_add(GTK_CONTAINER(toolItem), button);
371  gtk_widget_show(button);
372  if (rtl)
373  gtk_widget_set_margin_left(toolItem, 6);
374  else
375  gtk_widget_set_margin_right(toolItem, 6);
376 
377  gtk_container_add(GTK_CONTAINER(toolbar), toolItem);
378  gtk_widget_show(toolItem);
379  }
380 
381  button = createButton(toolbar, "printer-symbolic", _("Print document"), G_CALLBACK(printDocument));
382  toolItem = GTK_WIDGET(gtk_tool_item_new());
383  gtk_container_add(GTK_CONTAINER(toolItem), button);
384  gtk_widget_show(button);
385 
386  gtk_container_add(GTK_CONTAINER(toolbar), toolItem);
387  gtk_widget_show(toolItem);
388 }
