346 static void
347 job_finished_callback (EvJobAnnots *job,
348  EvSidebarAnnotations *sidebar_annots)
349 {
350  EvSidebarAnnotationsPrivate *priv;
351  GtkTreeStore *model;
352  GtkTreeSelection *selection;
353  GList *l;
354  GdkPixbuf *text_icon = NULL;
355  GdkPixbuf *attachment_icon = NULL;
356 
357  priv = sidebar_annots->priv;
358 
359  if (!job->annots) {
360  GtkTreeModel *list;
361 
362  list = ev_sidebar_annotations_create_simple_model (_("Document contains no annotations"));
363  gtk_tree_view_set_model (GTK_TREE_VIEW (priv->tree_view), list);
364  g_object_unref (list);
365 
366  g_object_unref (job);
367  priv->job = NULL;
368 
369  return;
370  }
371 
372  selection = gtk_tree_view_get_selection (GTK_TREE_VIEW (priv->tree_view));
373  gtk_tree_selection_set_mode (selection, GTK_SELECTION_SINGLE);
374  if (priv->selection_changed_id == 0) {
375  priv->selection_changed_id =
376  g_signal_connect (selection, "changed",
377  G_CALLBACK (selection_changed_cb),
378  sidebar_annots);
379  }
380 
381  model = gtk_tree_store_new (N_COLUMNS,
382  G_TYPE_STRING,
383  GDK_TYPE_PIXBUF,
384  G_TYPE_POINTER);
385 
386  for (l = job->annots; l; l = g_list_next (l)) {
387  EvMappingList *mapping_list;
388  GList *ll;
389  gchar *page_label;
390  GtkTreeIter iter;
391  gboolean found = FALSE;
392 
393  mapping_list = (EvMappingList *)l->data;
394  page_label = g_strdup_printf (_("Page %d"),
395  ev_mapping_list_get_page (mapping_list) + 1);
396  gtk_tree_store_append (model, &iter, NULL);
397  gtk_tree_store_set (model, &iter,
398  COLUMN_MARKUP, page_label,
399  -1);
400  g_free (page_label);
401 
402  for (ll = ev_mapping_list_get_list (mapping_list); ll; ll = g_list_next (ll)) {
403  EvAnnotation *annot;
404  const gchar *label;
405  const gchar *modified;
406  gchar *markup;
407  GtkTreeIter child_iter;
408  GdkPixbuf *pixbuf = NULL;
409 
410  annot = ((EvMapping *)(ll->data))->data;
411  if (!EV_IS_ANNOTATION_MARKUP (annot))
412  continue;
413 
414  label = ev_annotation_markup_get_label (EV_ANNOTATION_MARKUP (annot));
415  modified = ev_annotation_get_modified (annot);
416  if (modified) {
417  markup = g_strdup_printf ("<span weight=\"bold\">%s</span>\n%s",
418  label, modified);
419  } else {
420  markup = g_strdup_printf ("<span weight=\"bold\">%s</span>", label);
421  }
422 
423  if (EV_IS_ANNOTATION_TEXT (annot)) {
424  if (!text_icon) {
425  /* FIXME: use a better icon than EDIT */
426  text_icon = gtk_widget_render_icon_pixbuf (priv->tree_view,
427  GTK_STOCK_EDIT,
428  GTK_ICON_SIZE_BUTTON);
429  }
430  pixbuf = text_icon;
431  } else if (EV_IS_ANNOTATION_ATTACHMENT (annot)) {
432  if (!attachment_icon) {
433  attachment_icon = gtk_widget_render_icon_pixbuf (priv->tree_view,
434  EV_STOCK_ATTACHMENT,
435  GTK_ICON_SIZE_BUTTON);
436  }
437  pixbuf = attachment_icon;
438  }
439 
440  gtk_tree_store_append (model, &child_iter, &iter);
441  gtk_tree_store_set (model, &child_iter,
442  COLUMN_MARKUP, markup,
443  COLUMN_ICON, pixbuf,
444  COLUMN_ANNOT_MAPPING, ll->data,
445  -1);
446  g_free (markup);
447  found = TRUE;
448  }
449 
450  if (!found)
451  gtk_tree_store_remove (model, &iter);
452  }
453 
454  gtk_tree_view_set_model (GTK_TREE_VIEW (priv->tree_view),
455  GTK_TREE_MODEL (model));
456  g_object_unref (model);
457 
458  if (text_icon)
459  g_object_unref (text_icon);
460  if (attachment_icon)
461  g_object_unref (attachment_icon);
462 
463  g_object_unref (job);
464  priv->job = NULL;
465 }

468 ev_sidebar_annotations_load (EvSidebarAnnotations *sidebar_annots)

471
