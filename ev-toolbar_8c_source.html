152 
153  /* Set the MENUBAR style class so it's possible to drag the app

155  gtk_style_context_add_class (gtk_widget_get_style_context (GTK_WIDGET (ev_toolbar)),
156  GTK_STYLE_CLASS_MENUBAR);

160 
161  /* Navigation */

165  button = ev_toolbar_create_button (ev_toolbar, action);

302 gboolean
303 ev_toolbar_has_visible_popups (EvToolbar *ev_toolbar)
304 {
305  GtkAction *action;
306  GtkActionGroup *action_group;
307  GtkMenu *popup_menu;
308  EvToolbarPrivate *priv;
309 
310  g_return_val_if_fail (EV_IS_TOOLBAR (ev_toolbar), FALSE);
311 
312  priv = ev_toolbar->priv;
313 
314  popup_menu = gtk_menu_button_get_popup (GTK_MENU_BUTTON (priv->view_menu_button));
315  if (gtk_widget_get_visible (GTK_WIDGET (popup_menu)))
316  return TRUE;
317 
318  popup_menu = gtk_menu_button_get_popup (GTK_MENU_BUTTON (priv->action_menu_button));
319  if (gtk_widget_get_visible (GTK_WIDGET (popup_menu)))
320  return TRUE;
321 
322  action_group = ev_window_get_main_action_group (ev_toolbar->priv->window);
323  action = gtk_action_group_get_action (action_group, "ViewZoom");
324  if (ev_zoom_action_get_popup_shown (EV_ZOOM_ACTION (action)))
325  return TRUE;
326 
327  action = gtk_action_group_get_action (action_group, "History");
328  if (ev_history_action_get_popup_shown (EV_HISTORY_ACTION (action)))
329  return TRUE;
330 
331  return FALSE;
332 }
