92 void
93 ev_history_action_set_history (EvHistoryAction *action,
94  EvHistory *history)
95 {
96  GSList *proxies, *l;
97 
98  g_return_if_fail (EV_IS_HISTORY_ACTION (action));
99  g_return_if_fail (EV_IS_HISTORY (history));
100 
101  if (action->priv->history == history)
102  return;
103 
104  action->priv->history = history;
105  proxies = gtk_action_get_proxies (GTK_ACTION (action));
106  for (l = proxies; l && l->data; l = g_slist_next (l)) {
107  if (EV_IS_HISTORY_ACTION_WIDGET (l->data))
108  ev_history_action_widget_set_history (EV_HISTORY_ACTION_WIDGET (l->data), history);
109  }
110 }
