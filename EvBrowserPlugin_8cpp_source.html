107 template <typename IntegerType>
108 static inline void parseInteger(const char *strValue, IntegerType &intValue)
109 {
110  static const IntegerType intMax = std::numeric_limits<IntegerType>::max();
111  static const bool isSigned = std::numeric_limits<IntegerType>::is_signed;
112 
113  if (!strValue)
114  return;
115 
116  char *endPtr = nullptr;
117  errno = 0;
118  gint64 value = isSigned ? g_ascii_strtoll(strValue, &endPtr, 0) : g_ascii_strtoull(strValue, &endPtr, 0);
119  if (endPtr != strValue && errno == 0 && value <= intMax)
120  intValue = static_cast<IntegerType>(value);
121 }

123 static inline void parseDouble(const char *strValue, double &doubleValue)
124 {
125  if (!strValue)
126  return;
127 
128  char *endPtr = nullptr;
129  errno = 0;
130  double value = g_ascii_strtod(strValue, &endPtr);
131  if (endPtr != strValue && errno == 0)
132  doubleValue = value;
133 }

135 static inline void parseBoolean(const char *strValue, bool &boolValue)
136 {
137  if (!strValue)
138  return;
139 
140  unique_gptr<char> value(g_ascii_strdown(strValue, -1));
141  if (g_ascii_strcasecmp(value.get(), "false") == 0 || g_ascii_strcasecmp(value.get(), "no") == 0)
142  boolValue = false;
143  else if (g_ascii_strcasecmp(value.get(), "true") == 0 || g_ascii_strcasecmp(value.get(), "yes") == 0)
144  boolValue = true;
145  else {
146  int intValue = boolValue;
147  parseInteger<int>(strValue, intValue);
148  boolValue = intValue > 0;
149  }
150 }

152 static inline void parseZoomMode(const char *strValue, EvSizingMode &sizingModeValue)
153 {
154  if (!strValue)

158  if (g_ascii_strcasecmp(value.get(), "none") == 0)

165  sizingModeValue = EV_SIZING_AUTOMATIC;

168 NPError EvBrowserPlugin::initialize(NPMIMEType, uint16_t mode, int16_t argc, char *argn[], char *argv[], NPSavedData *)
169 {
170  // Default values.
171  bool toolbarVisible = true;
172  unsigned currentPage = 1;
173  EvSizingMode sizingMode = EV_SIZING_AUTOMATIC;
174  bool continuous = true;
175  bool dual = false;
176  double zoom = 0;
177 
178  for (int16_t i = 0; i < argc; ++i) {
179  if (g_ascii_strcasecmp(argn[i], "toolbar") == 0)
180  parseBoolean(argv[i], toolbarVisible);
181  else if (g_ascii_strcasecmp(argn[i], "currentpage") == 0)
182  parseInteger<unsigned>(argv[i], currentPage);
183  else if (g_ascii_strcasecmp(argn[i], "zoom") == 0)
184  parseDouble(argv[i], zoom);
185  else if (g_ascii_strcasecmp(argn[i], "zoommode") == 0)
186  parseZoomMode(argv[i], sizingMode);
187  else if (g_ascii_strcasecmp(argn[i], "continuous") == 0)
188  parseBoolean(argv[i], continuous);
189  else if (g_ascii_strcasecmp(argn[i], "dual") == 0)
190  parseBoolean(argv[i], dual);
191  }
192 
193  m_model = ev_document_model_new();
194  if (currentPage > 0)
195  ev_document_model_set_page(m_model, currentPage - 1);
196  ev_document_model_set_continuous(m_model, continuous);
197  ev_document_model_set_page_layout(m_model, dual ? EV_PAGE_LAYOUT_DUAL : EV_PAGE_LAYOUT_SINGLE);
198  if (zoom) {
199  ev_document_model_set_sizing_mode(m_model, EV_SIZING_FREE);
200  ev_document_model_set_scale(m_model, zoom);
201  } else
202  ev_document_model_set_sizing_mode(m_model, sizingMode);
203 
204  m_view = EV_VIEW(ev_view_new());
205  ev_view_set_model(m_view, m_model);
206 
207  m_toolbar = ev_browser_plugin_toolbar_new(this);
208  if (toolbarVisible)
209  gtk_widget_show(m_toolbar);
210 
211  return NPERR_NO_ERROR;
212 }

252 void EvBrowserPlugin::streamAsFile(NPStream *, const char *fname)
253 {
254  GFile *file = g_file_new_for_commandline_arg(fname);
255  unique_gptr<char> uri(g_file_get_uri(file));
256  g_object_unref(file);
257 
258  // Load the document synchronously here because the temporary file created by the browser
259  // is deleted when this function returns.
260  GError *error = nullptr;
261  EvDocument *document = ev_document_factory_get_document(uri.get(), &error);
262  if (!document) {
263  g_printerr("Error loading document %s: %s\n", uri.get(), error->message);
264  g_error_free(error);
265  } else {
266  ev_document_model_set_document(m_model, document);
267  g_object_unref(document);
268 
269  ev_view_set_loading(EV_VIEW(m_view), FALSE);
270  }

427 void EvBrowserPlugin::print() const
428 {
429  g_return_if_fail(EV_IS_DOCUMENT_MODEL(m_model));
430 
431  EvDocument *document = ev_document_model_get_document(m_model);
432  if (!document)
433  return;
434 
435  EvPrintOperation *printOperation = ev_print_operation_new(document);
436  if (!printOperation)
437  return;
438 
439  unique_gptr<char> outputBasename(g_path_get_basename(m_url.get()));
440  if (char *dot = g_strrstr(outputBasename.get(), "."))
441  dot[0] = '\0';
442 
443  unique_gptr<char> unescapedBasename(g_uri_unescape_string(outputBasename.get(), nullptr));
444  // Set output basename for printing to file.
445  GtkPrintSettings *printSettings = gtk_print_settings_new();
446  gtk_print_settings_set(printSettings, GTK_PRINT_SETTINGS_OUTPUT_BASENAME, unescapedBasename.get());
447 
448  if (const char *title = ev_document_get_title(document))
449  ev_print_operation_set_job_name(printOperation, title);
450  ev_print_operation_set_current_page(printOperation, ev_document_model_get_page(m_model));
451  ev_print_operation_set_embed_page_setup (printOperation, TRUE);
452  ev_print_operation_set_print_settings(printOperation, printSettings);
453  g_object_unref(printSettings);
454 
455  g_signal_connect(printOperation, "done", G_CALLBACK(g_object_unref), nullptr);
456 
457  GtkWidget *toplevel = gtk_widget_get_toplevel(GTK_WIDGET(m_view));
458  ev_print_operation_run(printOperation, GTK_IS_WINDOW(toplevel) ? GTK_WINDOW(toplevel) : nullptr);
459 }

509 bool EvBrowserPlugin::invoke(NPObject *npObject, NPIdentifier name, const NPVariant *args, uint32_t argCount, NPVariant *result)
510 {
511  EvBrowserPlugin *plugin = static_cast<EvBrowserPlugin *>(npObject);
512 
513  if (name == s_pluginClass.methodIdentifiers[EvBrowserPluginClass::Methods::GoToPage]) {
514  if (argCount != 1)
515  return false;
516 
517  if (NPVARIANT_IS_DOUBLE(args[0]))
518  plugin->goToPage(static_cast<unsigned>(NPVARIANT_TO_DOUBLE(args[0])));
519  else if (NPVARIANT_IS_STRING(args[0])) {
520  unique_gptr<char> pageLabel(g_strndup(NPVARIANT_TO_STRING(args[0]).UTF8Characters, NPVARIANT_TO_STRING(args[0]).UTF8Length));
521  plugin->goToPage(pageLabel.get());
522  } else
523  return false;
524 
525  VOID_TO_NPVARIANT(*result);
526  return true;
527  }
528  if (name == s_pluginClass.methodIdentifiers[EvBrowserPluginClass::Methods::ToggleContinuous]) {
529  plugin->toggleContinuous();
530  VOID_TO_NPVARIANT(*result);
531  return true;
532  }
533  if (name == s_pluginClass.methodIdentifiers[EvBrowserPluginClass::Methods::ToggleDual]) {
534  plugin->toggleDual();
535  VOID_TO_NPVARIANT(*result);
536  return true;
537  }
538  if (name == s_pluginClass.methodIdentifiers[EvBrowserPluginClass::Methods::ZoomIn]) {
539  plugin->zoomIn();
540  VOID_TO_NPVARIANT(*result);
541  return true;
542  }
543  if (name == s_pluginClass.methodIdentifiers[EvBrowserPluginClass::Methods::ZoomOut]) {
544  plugin->zoomOut();
545  VOID_TO_NPVARIANT(*result);
546  return true;
547  }
548  if (name == s_pluginClass.methodIdentifiers[EvBrowserPluginClass::Methods::Download]) {
549  plugin->download();
550  VOID_TO_NPVARIANT(*result);
551  return true;
552  }
553  if (name == s_pluginClass.methodIdentifiers[EvBrowserPluginClass::Methods::Print]) {
554  plugin->print();
555  VOID_TO_NPVARIANT(*result);
556  return true;
557  }
558  return false;
559 }

570 bool EvBrowserPlugin::getProperty(NPObject *npObject, NPIdentifier name, NPVariant *value)
571 {
572  EvBrowserPlugin *plugin = static_cast<EvBrowserPlugin *>(npObject);
573 
574  if (name == s_pluginClass.propertyIdentifiers[EvBrowserPluginClass::Properties::CurrentPage]) {
575  INT32_TO_NPVARIANT(plugin->currentPage() + 1, *value);
576  return true;
577  }
578  if (name == s_pluginClass.propertyIdentifiers[EvBrowserPluginClass::Properties::PageCount]) {
579  INT32_TO_NPVARIANT(plugin->pageCount(), *value);
580  return true;
581  }
582  if (name == s_pluginClass.propertyIdentifiers[EvBrowserPluginClass::Properties::Zoom]) {
583  DOUBLE_TO_NPVARIANT(plugin->zoom(), *value);
584  return true;
585  }
586  if (name == s_pluginClass.propertyIdentifiers[EvBrowserPluginClass::Properties::ZoomMode]) {
587  const char *zoomMode;
588 
589  switch (plugin->sizingMode()) {
590  case EV_SIZING_FREE:
591  zoomMode = "none";
592  break;
593  case EV_SIZING_FIT_PAGE:
594  zoomMode = "fit-page";
595  break;
596  case EV_SIZING_FIT_WIDTH:
597  zoomMode = "fit-width";
598  break;
599  case EV_SIZING_AUTOMATIC:
600  zoomMode = "auto";
601  break;
602  default:
603  return false;
604  }
605 
606  size_t zoomModeLength = strlen(zoomMode);
607  char *result = static_cast<char *>(NPN_MemAlloc(zoomModeLength + 1));
608  memcpy(result, zoomMode, zoomModeLength);
609  result[zoomModeLength] = '\0';
610 
611  STRINGZ_TO_NPVARIANT(result, *value);
612 
613  return true;
614  }
615  if (name == s_pluginClass.propertyIdentifiers[EvBrowserPluginClass::Properties::Continuous]) {
616  BOOLEAN_TO_NPVARIANT(plugin->isContinuous(), *value);
617  return true;
618  }
619  if (name == s_pluginClass.propertyIdentifiers[EvBrowserPluginClass::Properties::Dual]) {
620  BOOLEAN_TO_NPVARIANT(plugin->isDual(), *value);
